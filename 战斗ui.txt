  for(let i=0;i<25;i++) particles.push(createParticle());
  function draw(){
    ctx.clearRect(0,0,canvas.width,canvas.height);
    for(let i=particles.length-1;i>=0;i--){
      const p=particles[i];
      p.x+=p.vx; p.y+=p.vy; p.life--;
      if(p.life<=0||p.y<-10||p.x<-10||p.x>canvas.width+10){
        particles[i]=createParticle(); particles[i].y=canvas.height+5; continue;
      }
      const fade=Math.min(1,p.life/30);
      ctx.beginPath(); ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
      ctx.fillStyle=`hsla(42,58%,55%,${p.a*fade})`;
      ctx.fill();
    }
    raf=requestAnimationFrame(draw);
  }
  draw();
  window._stopParticles=()=>{cancelAnimationFrame(raf);};
})();

const FxEngine={
  canvas:null,ctx:null,pool:[],raf:null,
  init(){
    this.canvas=document.getElementById('fx-canvas');
    if(!this.canvas)return;
    this.ctx=this.canvas.getContext('2d');
    this.resize();
    window.addEventListener('resize',()=>this.resize());
  },
  resize(){
    if(!this.canvas)return;
    this.canvas.width=window.innerWidth;
    this.canvas.height=window.innerHeight;
  },
  spawn(inst){
    this.pool.push(inst);
    if(!this.raf)this._loop();
  },
  _loop(){
    const ctx=this.ctx;
    if(!ctx)return;
    ctx.clearRect(0,0,this.canvas.width,this.canvas.height);
    ctx.save();
    ctx.globalCompositeOperation='lighter';
    for(let i=this.pool.length-1;i>=0;i--){
      this.pool[i].tick(performance.now(),ctx);
      if(this.pool[i].done)this.pool.splice(i,1);
    }
    ctx.restore();
    if(this.pool.length)this.raf=requestAnimationFrame(()=>this._loop());
    else{ctx.clearRect(0,0,this.canvas.width,this.canvas.height);this.raf=null;}
  },
  clear(){
    this.pool.length=0;
    if(this.raf){cancelAnimationFrame(this.raf);this.raf=null;}
    this.ctx?.clearRect(0,0,this.canvas.width,this.canvas.height);
  }
};

function _hsla(h,s,l,a){return `hsla(${h},${s}%,${l}%,${a})`;}

class FxBurst{
  constructor({cx,cy,count=20,color=[42,80,65],speed=3,size=2,life=500,gravity=0,freeze=false,delay=0}){
    this.done=false;this.t0=performance.now()+delay;this.life=life;this.freeze=freeze;this.frozen=false;
    const[h,s,l]=color;this.h=h;this.s=s;this.l=l;
    this.ps=Array.from({length:count},()=>{
      const a=Math.random()*Math.PI*2,v=speed*(0.4+Math.random()*0.6);
      return{x:cx,y:cy,vx:Math.cos(a)*v,vy:Math.sin(a)*v,r:size*(0.4+Math.random()*0.6),a:0.7+Math.random()*0.3};
    });
    this.gravity=gravity;
  }
  tick(now,ctx){
    const el=now-this.t0;if(el<0)return;
    const pr=el/this.life;if(pr>=1){this.done=true;return;}
    const fade=1-pr;
    if(this.freeze&&pr>0.4&&!this.frozen){this.frozen=true;this.ps.forEach(p=>{p.vx*=0.05;p.vy*=0.05;});}
    for(const p of this.ps){
      p.x+=p.vx;p.y+=p.vy;p.vy+=this.gravity;
      const r=p.r*Math.max(0.2,fade);
      ctx.beginPath();ctx.arc(p.x,p.y,r,0,Math.PI*2);
      ctx.fillStyle=_hsla(this.h,this.s,this.l,fade*p.a);
      ctx.fill();
    }
  }
}

class FxBeam{
  constructor({from,to,color=[200,70,75],width=4,life=400,delay=0}){
    this.done=false;this.t0=performance.now()+delay;this.life=life;
    this.fx=from.x;this.fy=from.y;this.tx=to.x;this.ty=to.y;
    const[h,s,l]=color;this.h=h;this.s=s;this.l=l;this.w=width;
  }
  tick(now,ctx){
    const el=now-this.t0;if(el<0)return;
    const pr=el/this.life;if(pr>=1){this.done=true;return;}
    const headPr=Math.min(1,pr*2.5);
    const tailPr=Math.max(0,(pr-0.3)/0.7);
    const hx=this.fx+(this.tx-this.fx)*headPr,hy=this.fy+(this.ty-this.fy)*headPr;
    const tx=this.fx+(this.tx-this.fx)*tailPr,ty=this.fy+(this.ty-this.fy)*tailPr;
    const fade=pr<0.7?1:(1-(pr-0.7)/0.3);
    ctx.save();
    ctx.strokeStyle=_hsla(this.h,this.s,this.l,fade*0.9);
    ctx.lineWidth=this.w;ctx.lineCap='round';
    ctx.shadowColor=_hsla(this.h,this.s,this.l+15,fade*0.6);ctx.shadowBlur=12;
    ctx.beginPath();ctx.moveTo(tx,ty);ctx.lineTo(hx,hy);ctx.stroke();
    // 头部亮点
    ctx.beginPath();ctx.arc(hx,hy,this.w*1.5,0,Math.PI*2);
    ctx.fillStyle=_hsla(this.h,this.s,Math.min(95,this.l+25),fade*0.8);ctx.fill();
    ctx.restore();
  }
}

class FxTrail{
  constructor({from,to,color=[45,90,55],width=6,particles=20,life=500,delay=0,curve=0}){
    this.done=false;this.t0=performance.now()+delay;this.life=life;
    const[h,s,l]=color;this.h=h;this.s=s;this.l=l;this.w=width;
    // 贝塞尔控制点
    const mx=(from.x+to.x)/2,my=(from.y+to.y)/2;
    const dx=to.x-from.x,dy=to.y-from.y;
    this.cp={x:mx+(-dy)*curve,y:my+(dx)*curve};
    this.from=from;this.to=to;
    this.ps=Array.from({length:particles},(_,i)=>({
      t:i/particles,off:{x:(Math.random()-0.5)*8,y:(Math.random()-0.5)*8},
      r:width*(0.3+Math.random()*0.7),a:0.5+Math.random()*0.5
    }));
  }
  _bezier(t){
    const u=1-t;
    return{
      x:u*u*this.from.x+2*u*t*this.cp.x+t*t*this.to.x,
      y:u*u*this.from.y+2*u*t*this.cp.y+t*t*this.to.y
    };
  }
  tick(now,ctx){
    const el=now-this.t0;if(el<0)return;
    const pr=el/this.life;if(pr>=1){this.done=true;return;}
    const head=Math.min(1,pr*1.8);
    for(const p of this.ps){
      const pt=p.t*head;if(pt>head)continue;
      const localFade=1-Math.abs(pt-head*0.8)/Math.max(0.01,head);
      const pos=this._bezier(pt);
      ctx.beginPath();ctx.arc(pos.x+p.off.x,pos.y+p.off.y,p.r*Math.max(0.1,localFade),0,Math.PI*2);
      ctx.fillStyle=_hsla(this.h,this.s,this.l,Math.max(0,localFade)*p.a*0.8);
      ctx.fill();
    }
    // 头部亮点
    const hp=this._bezier(head);
    ctx.beginPath();ctx.arc(hp.x,hp.y,this.w*2,0,Math.PI*2);
    ctx.fillStyle=_hsla(this.h,this.s,Math.min(95,this.l+20),0.7*(1-pr));
    ctx.fill();
  }
}

class FxFistProjectile{
  constructor({from,to,color=[6,70,55],size=38,life=420,delay=0}){
    this.done=false;this.t0=performance.now()+delay;this.life=life;
    this.from=from;this.to=to;this.size=size;
    const[h,s,l]=color;this.h=h;this.s=s;this.l=l;
    this.img=G.fistImg||null;
    this.trail=[];this._spawned=false;
    // 飞行角度：SVG拳头自然朝上(-90°)，需补偿+90°让拳面对准目标
    this.angle=Math.atan2(to.y-from.y,to.x-from.x)+Math.PI/2;
  }
  tick(now,ctx){
    const el=now-this.t0;if(el<0)return;
    const pr=Math.min(1,el/this.life);
    if(pr>=1){
      // 命中时生成扩散波 + 粒子爆发
      if(!this._spawned){
        this._spawned=true;
        FxEngine.spawn(new FxRing({cx:this.to.x,cy:this.to.y,radius:70,color:[this.h,this.s,this.l],width:3,life:450}));
        FxEngine.spawn(new FxRing({cx:this.to.x,cy:this.to.y,radius:45,color:[this.h,this.s,Math.min(90,this.l+15)],width:2,life:350}));
        FxEngine.spawn(new FxBurst({cx:this.to.x,cy:this.to.y,count:30,color:[this.h,this.s+10,this.l],speed:4.5,size:2.5,life:400}));
      }
      this.done=true;return;
    }
    // 缓入缓出飞行曲线
    const t=pr<0.5?2*pr*pr:1-Math.pow(-2*pr+2,2)/2;
    const x=this.from.x+(this.to.x-this.from.x)*t;
    const y=this.from.y+(this.to.y-this.from.y)*t;
    // 拖尾记录
    this.trail.push({x,y,t:now});
    if(this.trail.length>12)this.trail.shift();
    // 绘制拖尾
    for(let i=0;i<this.trail.length-1;i++){
      const tp=this.trail[i];
      const fade=(i+1)/this.trail.length*0.4;
      const sz=this.size*(0.3+fade*0.5);
      ctx.beginPath();ctx.arc(tp.x,tp.y,sz*0.25,0,Math.PI*2);
      ctx.fillStyle=_hsla(this.h,this.s,this.l+20,fade);
      ctx.fill();
    }
    // 绘制拳头
    const drawSz=this.size*(0.8+pr*0.3);
    ctx.save();
    ctx.translate(x,y);
    ctx.rotate(this.angle);
    // 外发光
    ctx.shadowColor=_hsla(this.h,this.s,this.l+10,0.6);
    ctx.shadowBlur=18;
    if(this.img&&this.img.complete){
      ctx.drawImage(this.img,-drawSz/2,-drawSz/2,drawSz,drawSz);
      // 二次绘制加强发光
      ctx.globalAlpha=0.3;
      ctx.shadowBlur=30;
      ctx.drawImage(this.img,-drawSz/2,-drawSz/2,drawSz,drawSz);
      ctx.globalAlpha=1;
    } else {
      // fallback: 简笔圆拳
      ctx.beginPath();ctx.arc(0,0,drawSz*0.35,0,Math.PI*2);
      ctx.fillStyle=_hsla(this.h,this.s,this.l,0.8);
      ctx.fill();
    }
    ctx.restore();
  }
}

// ── 剑斩特效（手柄在左为圆心，剑尖在右划出弧线） ──
class FxSwordSlash{
  constructor({cx,cy,color=[200,60,70],size=100,life=450,delay=0}){
    this.done=false;this.t0=performance.now()+delay;this.life=life;
    this.cx=cx;this.cy=cy;this.size=size;
    // 剑尖在右，手柄在左。圆心=手柄端，在目标左上方
    this.pivotX=cx-size*0.45;this.pivotY=cy-size*0.15;
    const[h,s,l]=color;this.h=h;this.s=s;this.l=l;
    this.img=G.swordImg||null;
    this._spawned=false;
    this.startAngle=-35*Math.PI/180;
    this.endAngle=35*Math.PI/180;
    this.arcTrail=[];
  }
  tick(now,ctx){
    const el=now-this.t0;if(el<0)return;
    const pr=Math.min(1,el/this.life);
    if(pr>=1){
      if(!this._spawned){this._spawned=true;
        FxEngine.spawn(new FxBurst({cx:this.cx,cy:this.cy,count:22,color:[this.h,this.s,this.l],speed:4,size:2,life:350}));
      } this.done=true;return;
    }
    const fadeIn=Math.min(1,pr/0.12),fadeOut=pr>0.6?1-(pr-0.6)/0.4:1;
    const alpha=fadeIn*fadeOut;
    const slashPr=clamp((pr-0.08)/0.5,0,1);
    const eased=slashPr<0.5?4*slashPr*slashPr*slashPr:1-Math.pow(-2*slashPr+2,3)/2;
    const angle=this.startAngle+(this.endAngle-this.startAngle)*eased;
    const scale=1+Math.sin(slashPr*Math.PI)*0.12;
    const drawW=this.size*scale,drawH=this.size*0.32*scale;
    ctx.save();
    ctx.translate(this.pivotX,this.pivotY);
    ctx.rotate(angle);
    ctx.globalAlpha=alpha;
    ctx.shadowColor=_hsla(this.h,this.s,this.l+15,alpha*0.7);ctx.shadowBlur=22;
    if(this.img&&this.img.complete){
      // 剑身从圆心向右延伸，剑尖朝右对着目标
      ctx.drawImage(this.img,0,-(drawH/2),drawW,drawH);
      ctx.globalAlpha=alpha*0.25;ctx.shadowBlur=40;
      ctx.drawImage(this.img,0,-(drawH/2),drawW,drawH);
    } else {
      ctx.fillStyle=_hsla(this.h,this.s,this.l,alpha*0.8);
      ctx.fillRect(0,-2,drawW,4);
    }
    ctx.restore();
    // 剑尖弧线拖影（剑尖在圆心右侧 drawW 处）
    if(slashPr>0.02&&slashPr<0.95){
      const tipX=this.pivotX+Math.cos(angle)*drawW;
      const tipY=this.pivotY+Math.sin(angle)*drawW;
      this.arcTrail.push({x:tipX,y:tipY,a:alpha});

// ── 降龙十八掌 · 龙飞行特效（龙头朝向目标，身体蛇形扭动飞行） ──
class FxDragonFlight{
  constructor({from,to,color=[42,80,55],size=220,life=750,delay=0}){
    this.done=false;this.t0=performance.now()+delay;this.life=life;
    this.from=from;this.to=to;this.size=size;
    const[h,s,l]=color;this.h=h;this.s=s;this.l=l;
    this.img=G.dragonImg||null;
    this._spawned=false;
    this.angle=Math.atan2(to.y-from.y,to.x-from.x);
    this.dist=Math.hypot(to.x-from.x,to.y-from.y);
    // 龙SVG自然朝向：龙头在左上角，身体蜿蜒到右下
    // 龙头面朝方向约为左上 ≈ -150°（即 atan2(-1, -1.5) ≈ -2.55 rad）
    // 要让龙头对准飞行方向(this.angle)，需要补偿这个自然偏移
    this._svgHeadAngle=Math.atan2(-300,  -500); // 龙头朝左上的角度 ≈ -2.55 rad
  }
  tick(now,ctx){
    const el=now-this.t0;if(el<0)return;
    const pr=Math.min(1,el/this.life);
    if(pr>=1){
      if(!this._spawned){this._spawned=true;
        // 命中大爆发
        FxEngine.spawn(new FxRing({cx:this.to.x,cy:this.to.y,radius:100,color:[this.h,this.s,this.l],width:5,life:600}));
        FxEngine.spawn(new FxRing({cx:this.to.x,cy:this.to.y,radius:65,color:[this.h,this.s,this.l+10],width:3,life:500,delay:80}));
        FxEngine.spawn(new FxBurst({cx:this.to.x,cy:this.to.y,count:50,color:[this.h,this.s+10,this.l+5],speed:7,size:3.5,life:550}));
        FxEngine.spawn(new FxGlyph({cx:this.to.x,cy:this.to.y-35,char:'龙',color:[this.h,this.s,this.l],size:65,life:900}));
      }
      this.done=true;return;
    }
    // 飞行路径：加速冲向目标
    const t=pr*pr*(3-2*pr);
    const baseX=this.from.x+(this.to.x-this.from.x)*t;
    const baseY=this.from.y+(this.to.y-this.from.y)*t;
    // 蛇形扭动：身体沿垂直于飞行方向做正弦波动，频率随时间加快
    const wiggleAmp=35*(1-pr*0.6); // 越接近目标扭动越小
    const wiggleFreq=Math.PI*5; // 2.5个完整波
    const wiggle=Math.sin(pr*wiggleFreq)*wiggleAmp;
    const perpX=-Math.sin(this.angle)*wiggle;
    const perpY=Math.cos(this.angle)*wiggle;
    const x=baseX+perpX,y=baseY+perpY;
    // 龙身缩放：起飞时较小→飞行中饱满→接近目标时略收
    const scalePr=pr<0.15?pr/0.15:(pr>0.85?1-(pr-0.85)/0.15*0.2:1);
    const drawW=this.size*scalePr;
    const drawH=this.size*0.5*scalePr;
    // 龙头朝向：基础飞行角度 + 扭动带来的角度偏转
    const wiggleAngle=Math.cos(pr*wiggleFreq)*0.2*(1-pr*0.5);
    const fadeIn=Math.min(1,pr*5);
    const alpha=fadeIn*(pr>0.88?1-(pr-0.88)/0.12:1);
    ctx.save();
    ctx.translate(x,y);
    // 龙头朝向目标：SVG龙头在左上角，自然朝向约-150°
    // 旋转补偿 = 飞行方向 - 龙头自然朝向
    const rot=this.angle-this._svgHeadAngle+wiggleAngle;
    ctx.rotate(rot);
    ctx.globalAlpha=alpha;
    // 金色发光
    ctx.shadowColor=_hsla(this.h,this.s,this.l+15,0.9);ctx.shadowBlur=35;
    if(this.img&&this.img.complete){
      ctx.drawImage(this.img,-drawW/2,-drawH/2,drawW,drawH);
      // 二次绘制加强金色光芒
      ctx.globalAlpha=alpha*0.3;ctx.shadowBlur=60;
      ctx.drawImage(this.img,-drawW/2,-drawH/2,drawW,drawH);
    } else {
      ctx.fillStyle=_hsla(this.h,this.s,this.l,alpha*0.8);
      ctx.fillRect(-drawW/2,-drawH/4,drawW,drawH/2);
    }
    ctx.restore();
    // 龙尾火焰粒子拖尾
    if(pr>0.08){
      const tailDir=this.angle+Math.PI; // 尾巴方向=飞行反方向
      for(let i=0;i<5;i++){
        const spread=(Math.random()-0.5)*25;
        const dist=drawW*(0.3+Math.random()*0.2);
        const tx=x+Math.cos(tailDir)*dist+Math.cos(tailDir+Math.PI/2)*spread;
        const ty=y+Math.sin(tailDir)*dist+Math.sin(tailDir+Math.PI/2)*spread;
        const r=1.5+Math.random()*3;
        ctx.beginPath();ctx.arc(tx,ty,r,0,Math.PI*2);
        ctx.fillStyle=_hsla(this.h+Math.random()*30-15,this.s+10,this.l+25,alpha*0.4*Math.random());
        ctx.fill();
      }
    }
  }
}

          if(!kf||learned.includes(n))return false;
          const req=kf.基本功要求||{};
          for(const bk of BKEYS) if((req[bk]||0)>(bskNow[bk]||0)+10) return false;
          return true;
        });
        const preferred=allCands.filter(n=>{
          const req=KUNGFU_LIB[n].基本功要求||{};
          return (req[topBsk]||0)>0;
        });
        const fallback=allCands.filter(n=>!preferred.includes(n));
        // 打乱各组后合并：优先组在前
        const shuffle=arr=>{for(let i=arr.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[arr[i],arr[j]]=[arr[j],arr[i]];}return arr;};
        const candidates=[...shuffle(preferred),...shuffle(fallback)];
        const picks=candidates.slice(0,need);
        if(picks.length){
          learned=[...learned,...picks];
          e.领悟功法=learned;
          changed=true;
        }
      }
    }

    // ── 步骤1：汇总武功增幅（只读）──
    let kfHpBonus=0, kfMpBonus=0;
    for(const kfName of learned){
      const kf=KUNGFU_LIB[kfName]; if(!kf)continue;
      const b=kf.基本功和属性增幅||{};
      kfHpBonus+=(b.血量上限||0);
      kfMpBonus+=(b.内力上限||0);
    }

    // ── 步骤2：基本功补全（目标总值=eqLv×5）──
    const bsk=e.基本功||{};
    let targetTotal=lv*5;
    if(lv>=60) targetTotal+=250;
    else if(lv>=50) targetTotal+=150;
    else if(lv>=40) targetTotal+=100;
    else if(lv>=35) targetTotal+=60;
    else if(lv>=30) targetTotal+=30;
    let curTotal=0;
    for(const k of BKEYS) curTotal+=(bsk[k]||0);

    if(curTotal<targetTotal){
      changed=true;
      if(curTotal>0){
        // 按原比例等比放大
        const ratio=targetTotal/curTotal;
        let allocated=0;
        let maxK=BKEYS[0], maxV=0;
        for(const k of BKEYS){
          const v=F((bsk[k]||0)*ratio);
          bsk[k]=v; allocated+=v;
          if(v>maxV){maxV=v;maxK=k;}
        }
        // 余数分给最大项
        if(allocated<targetTotal) bsk[maxK]+=(targetTotal-allocated);
      } else {
        // 全零：按武功基本功要求推断比例
        const weights={};
        for(const k of BKEYS) weights[k]=0;
        for(const kfName of learned){
          const kf=KUNGFU_LIB[kfName]; if(!kf)continue;
          const req=kf.基本功要求||{};
          for(const k of BKEYS) weights[k]=Math.max(weights[k],req[k]||0);
        }
        let wTotal=0;
        for(const k of BKEYS) wTotal+=weights[k];
        if(wTotal>0){
          let allocated=0; let maxK=BKEYS[0], maxV=0;
          for(const k of BKEYS){
            const v=F(weights[k]/wTotal*targetTotal);
            bsk[k]=v; allocated+=v;
            if(v>maxV){maxV=v;maxK=k;}
          }
          if(allocated<targetTotal) bsk[maxK]+=(targetTotal-allocated);
        } else {
          // 无武功或武功无要求：平均分配
          const per=F(targetTotal/6);
          for(const k of BKEYS) bsk[k]=per;
          bsk[BKEYS[0]]+=(targetTotal-per*6);
        }
      }
      e.基本功=bsk;
    }

    // ── 步骤3：属性补全（境界等效成长+基本功贡献，逐项取max）──
    const attr=e.属性||{};
    const formulaAttr={力量:0,速度:0,防御力:0,内功修为:0,身法:0};
    for(const a of AKEYS) formulaAttr[a]+=F(lv*LV_GROWTH[a]);
    for(const k of BKEYS){
      const val=bsk[k]||0; if(!val)continue;
      const map=BSKILL_ATTR_MAP[k]; if(!map)continue;
      for(const[a,rate]of Object.entries(map)) formulaAttr[a]+=F(val*rate);
    }
    let attrChanged=false;
    for(const a of AKEYS){
      const old=attr[a]||0;
      if(formulaAttr[a]>old){ attr[a]=formulaAttr[a]; attrChanged=true; }
    }
    if(attrChanged){ e.属性=attr; changed=true; }

    // ── 步骤4：血量/内力上限补全 ──
    const core=e.核心状态||{};
    const hp=core.血量||{}, mp=core.内力||{};
    const oldHpMax=hp.上限||0, oldMpMax=mp.上限||0;
    const oldHpCur=hp.当前??oldHpMax, oldMpCur=mp.当前??oldMpMax;

    let bskSum=0; for(const k of BKEYS) bskSum+=(bsk[k]||0);
    const fHpMax=300+bskSum*8+(attr.力量||0)*20+kfHpBonus;
    const fMpMax=300+bskSum*8+(attr.内功修为||0)*20+kfMpBonus;
    const newHpMax=Math.max(fHpMax,oldHpMax);
    const newMpMax=Math.max(fMpMax,oldMpMax);

    if(newHpMax!==oldHpMax||newMpMax!==oldMpMax){
      changed=true;
      const hpRatio=oldHpMax>0?oldHpCur/oldHpMax:1;
      const mpRatio=oldMpMax>0?oldMpCur/oldMpMax:1;
      hp.上限=newHpMax;
      hp.当前=Math.max(1,Math.min(F(newHpMax*hpRatio),newHpMax));
      mp.上限=newMpMax;
      mp.当前=Math.max(0,Math.min(F(newMpMax*mpRatio),newMpMax));
      core.血量=hp; core.内力=mp;
      e.核心状态=core;
    }
  }

  // ── 步骤5：写回MVU ──
  if(changed){
    const Mvu=getMvu();
    if(Mvu) await Mvu.replaceMvuData(mvuData,{type:'message',message_id:G.aiMsgId});
    console.log('[战斗UI] 敌人属性补全完成');
  }
}

// ── 队友属性补全（公式值 vs MVU值，哪个高取哪个）─────────────
async function fixAllyStats(mvuData){
  const allies=mvuData.stat_data?.我方队友;
  if(!allies||!Object.keys(allies).length)return;
  const F=Math.floor;
  const BKEYS=['拳掌','棍枪','剑','刀','暗器','内功底蕴'];
  const AKEYS=['力量','速度','防御力','内功修为','身法'];

    拳掌:{力量:0.5,防御力:0.2},棍枪:{力量:0.3,速度:0.3},
    剑:{速度:0.4,身法:0.3},刀:{力量:0.4,防御力:0.3},
    暗器:{速度:0.4,身法:0.4},内功底蕴:{内功修为:0.8,防御力:0.2},
  };
  const LV_GROWTH={力量:0.8,速度:0.8,防御力:0.6,内功修为:1.0,身法:0.6};
  let changed=false;

  for(const[key,a]of Object.entries(allies)){

    const lv=realmToEqLv(realm);
    const learned=Array.isArray(a.领悟功法)?a.领悟功法:[];

    // ── 步骤1：汇总武功增幅 ──
    let kfHpBonus=0, kfMpBonus=0;
    for(const kfName of learned){
      const kf=KUNGFU_LIB[kfName]; if(!kf)continue;
      const b=kf.基本功和属性增幅||{};
      kfHpBonus+=(b.血量上限||0);
      kfMpBonus+=(b.内力上限||0);
    }

    // ── 步骤2：基本功补全（目标总值=eqLv×5）──
    const bsk=a.基本功||{};
    const targetTotal=lv*5;
    let curTotal=0;
    for(const k of BKEYS) curTotal+=(bsk[k]||0);

    if(curTotal<targetTotal){
      changed=true;
      if(curTotal>0){
        const ratio=targetTotal/curTotal;
        let allocated=0;
        let maxK=BKEYS[0], maxV=0;
        for(const k of BKEYS){
          const v=F((bsk[k]||0)*ratio);
          bsk[k]=v; allocated+=v;
          if(v>maxV){maxV=v;maxK=k;}
        }
        if(allocated<targetTotal) bsk[maxK]+=(targetTotal-allocated);
      } else {
        const weights={};
        for(const k of BKEYS) weights[k]=0;
        for(const kfName of learned){
          const kf=KUNGFU_LIB[kfName]; if(!kf)continue;
          const req=kf.基本功要求||{};
          for(const k of BKEYS) weights[k]=Math.max(weights[k],req[k]||0);
        }
        let wTotal=0;
        for(const k of BKEYS) wTotal+=weights[k];
        if(wTotal>0){
          let allocated=0; let maxK=BKEYS[0], maxV=0;
          for(const k of BKEYS){
            const v=F(weights[k]/wTotal*targetTotal);
            bsk[k]=v; allocated+=v;
            if(v>maxV){maxV=v;maxK=k;}
          }
          if(allocated<targetTotal) bsk[maxK]+=(targetTotal-allocated);
        } else {
          const per=F(targetTotal/6);
          for(const k of BKEYS) bsk[k]=per;
          bsk[BKEYS[0]]+=(targetTotal-per*6);
        }
      }
      a.基本功=bsk;
    }

    // ── 步骤3：属性补全（境界等效成长+基本功贡献+武学属性增幅，逐项取max）──
    const attr=a.属性||{};
    const formulaAttr={力量:0,速度:0,防御力:0,内功修为:0,身法:0};
    for(const ak of AKEYS) formulaAttr[ak]+=F(lv*LV_GROWTH[ak]);
    for(const k of BKEYS){
      const val=bsk[k]||0; if(!val)continue;
      const map=BSKILL_ATTR_MAP[k]; if(!map)continue;
      for(const[ak,rate]of Object.entries(map)) formulaAttr[ak]+=F(val*rate);
    }
    // 武学属性增幅
    for(const kfName of learned){
      const kf=KUNGFU_LIB[kfName]; if(!kf)continue;
      const b=kf.基本功和属性增幅||{};
      for(const ak of AKEYS) formulaAttr[ak]+=(b[ak]||0);
    }
    let attrChanged=false;
    for(const ak of AKEYS){
      const old=attr[ak]||0;
      if(formulaAttr[ak]>old){ attr[ak]=formulaAttr[ak]; attrChanged=true; }
    }
    if(attrChanged){ a.属性=attr; changed=true; }

    // ── 步骤4：血量/内力上限补全 ──
    const core=a.核心状态||{};
    const hp=core.血量||{}, mp=core.内力||{};
    const oldHpMax=hp.上限||0, oldMpMax=mp.上限||0;
    const oldHpCur=hp.当前??oldHpMax, oldMpCur=mp.当前??oldMpMax;

    let bskSum=0; for(const k of BKEYS) bskSum+=(bsk[k]||0);
    const fHpMax=300+bskSum*8+(attr.力量||0)*20+kfHpBonus;
    const fMpMax=300+bskSum*8+(attr.内功修为||0)*20+kfMpBonus;
    const newHpMax=Math.max(fHpMax,oldHpMax);
    const newMpMax=Math.max(fMpMax,oldMpMax);

    if(newHpMax!==oldHpMax||newMpMax!==oldMpMax){
      changed=true;
      const hpRatio=oldHpMax>0?oldHpCur/oldHpMax:1;
      const mpRatio=oldMpMax>0?oldMpCur/oldMpMax:1;
      hp.上限=newHpMax;
      hp.当前=Math.max(1,Math.min(F(newHpMax*hpRatio),newHpMax));
      mp.上限=newMpMax;
      mp.当前=Math.max(0,Math.min(F(newMpMax*mpRatio),newMpMax));
      core.血量=hp; core.内力=mp;
      a.核心状态=core;
    }
  }

  if(changed){
    const Mvu=getMvu();
    if(Mvu) await Mvu.replaceMvuData(mvuData,{type:'message',message_id:G.aiMsgId});
    console.log('[战斗UI] 队友属性补全完成');
  }
}

async function init(){
  // 预加载SVG图像（从资源脚本）
  try{
    const res=window.parent.WX_RES?.svg||{};
    const _loadSvg=(svgStr)=>{
      if(!svgStr)return null;
      const img=new Image();
      img.src='data:image/svg+xml;charset=utf-8,'+encodeURIComponent(svgStr);
      return img;
    };
    G.fistImg=_loadSvg(res.fist);
    G.palmImg=_loadSvg(res.palm);
    G.swordImg=_loadSvg(res.sword);
    G.bladeImg=_loadSvg(res.blade);
    G.dragonImg=_loadSvg(res.dragon);
    G.spearImg=_loadSvg(res.spear);
    G.staffImg=_loadSvg(res.staff);
  }catch(e){}
  // 加载音效资源
  try{
    const sfxRes=window.parent.WX_RES?.sfx||{};
    if(sfxRes.fist)SFX.fist=sfxRes.fist;
    if(sfxRes.sword)SFX.sword=sfxRes.sword;
    if(sfxRes.spear)SFX.spear=sfxRes.spear;
    if(sfxRes.dart)SFX.dart=sfxRes.dart;
    if(sfxRes.dragon)SFX.dragon=sfxRes.dragon;
    if(sfxRes.acupoint)SFX.acupoint=sfxRes.acupoint;
    if(sfxRes.heal)SFX.heal=sfxRes.heal;
    if(sfxRes.defend)SFX.defend=sfxRes.defend;
  }catch(e){}
  // 预加载音效到AudioBuffer缓存
  Object.values(SFX).forEach(u=>{if(u)_decodeSfx(u);});
  await loadKungfuLib();
  // ── 注入自创武功到武功库 ──
  _injectCustomKungfu();
  log('══ 天书江湖录 · 战斗开启 ══','l-sys');
  const mvu=await loadMvu();
  const sd=mvu?.stat_data;
  if(!sd||(!sd.玩家&&!sd.敌人)){
    log('未读到有效战斗数据，请确认MVU已就绪并刷新','l-sys');
    return;
  }
  const useData=sd;
  // ── 敌人属性补全 ──
  if(useData.敌人&&Object.keys(useData.敌人).length){
    await fixEnemyStats(mvu);
    // 补全后直接用mvu.stat_data（已被fixEnemyStats原地修改并写回MVU）
  }
  // ── 队友属性补全 ──
  if(useData.我方队友&&Object.keys(useData.我方队友).length){
    await fixAllyStats(mvu);
  }
  G.actors={};G.pKeys=[];G.eKeys=[];G.statuses={};G.cds={};G.atbProg={};G.copiedKf={};
  G.defending=new Set();G.jiazhuMap={};G.roundCount=0;G.result=null;G.settleChoice=null;G.profGained=0;
  G.battleLog=[];G.itemsUsed={};

  if(useData.玩家){
    const k='player';
    G.actors[k]=buildActor(k,'player',useData.玩家);
    G.pKeys.push(k);
    G.statuses[k]=parseStatusList(useData.玩家.核心状态?.状态列表);
    G.cds[k]={};
  }
  for(const[name,charData]of Object.entries(useData.我方队友||{})){
    if(charData.是否参战===false) continue;
    G.actors[name]=buildActor(name,'ally',charData);
    G.pKeys.push(name);
    G.statuses[name]=parseStatusList(charData.核心状态?.状态列表);
    G.cds[name]={};
  }
  for(const[key,charData]of Object.entries(useData.敌人||{})){
    G.actors[key]=buildActor(key,'enemy',charData);
    G.eKeys.push(key);
    G.statuses[key]=parseStatusList(charData.核心状态?.状态列表);
    G.cds[key]={};
  }
  // ── 心魔战：敌人仅一个且名字以[心魔]开头 → 队友不参战 ──
  if(G.eKeys.length===1){
    const eName=G.actors[G.eKeys[0]]?.name||'';
    if(/^\[心魔\]/.test(eName)){
      for(const k of [...G.pKeys]){
        if(k!=='player'){
          delete G.actors[k];
          delete G.statuses[k];
          delete G.cds[k];
        }
      }
      G.pKeys=G.pKeys.filter(k=>k==='player');
    }
  }
  const playerItems = G.actors['player']?.items || {};
  for(const k of G.pKeys){ if(G.actors[k]) G.actors[k].playerItemsRef = playerItems; }
  G.itemUsageLimit = G.pKeys.length * 10;
  for(const k of allKeys()) G.atbProg[k]=Math.random()*25;

  // 检查倍速安全性：若任何角色在某倍速下单tick溢出，禁用该倍速
  for(const s of [2,4]){
    const overflow=allKeys().some(k=>{
      const spd=getAttr(k,'速度')||50;
      return (spd*0.05+1.5)*s>=100;
    });
    if(overflow){
      const btn=q(s===2?'b-spd2':'b-spd4');
      if(btn){btn.disabled=true;btn.style.opacity='0.3';}
    }
  }

  // ── 自动应用被动状态（影响回合===-1 的永久效果）──
  for(const k of allKeys()){
    const ac=gA(k); if(!ac)continue;
    for(const kfName of ac.kungfu||[]){
      const kf=KUNGFU_LIB[kfName]; if(!kf)continue;
      const eff=kf.影响状态; if(!eff||!eff.名称||eff.名称==='无')continue;
      // 内功类：仅主运内功且影响回合===-1（被动）才开局生效
      if(kf.类别==='内功'){
        if(kfName===ac.mainKf&&eff.影响回合===-1){
          applyStatus(k,{名称:eff.名称,数值:eff.数值||0,剩余回合:-1,来源:kfName});
          log(`${ac.name}【${kfName}】${eff.名称} 生效`,'l-buff');
        }
        continue;
      }
      // 非内功：影响回合===-1 则开局自动生效
      if(eff.影响回合===-1){
        applyStatus(k,{名称:eff.名称,数值:eff.数值||0,剩余回合:-1,来源:kfName});
        log(`${ac.name}【${kfName}】${eff.名称} 生效`,'l-buff');
      }
    }
  }

  renderAll();
  G.phase='atb';
  atbLoop();
  log(`战场就位。敌方：${G.eKeys.map(k=>G.actors[k]?.name).join('、')}（共${G.eKeys.length}人）`,'l-sys');
  // ── 实力压制 ──
  const _maxPPow=Math.max(...G.pKeys.map(k=>gA(k)?.powerScore||0));
  const _maxEPow=Math.max(...G.eKeys.map(k=>gA(k)?.powerScore||0));
  const _powGap=_maxEPow-_maxPPow;
  G._overLvTier=_powGap>=400?2:_powGap>=200?1:0;
  if(G._overLvTier){
    const spdMul=G._overLvTier===2?1.7:1.5;
    // 写入MVU
    try{
      const Mvu=getMvu();
      if(Mvu&&G.aiMsgId!==null){
        const mvd=await Mvu.getMvuData({type:'message',message_id:G.aiMsgId});
        const enemies=mvd?.stat_data?.敌人;
        if(enemies){
          let needWrite=false;
          for(const[ek,ed]of Object.entries(enemies)){
            if(ed.属性&&!ed._越级速度已修正){
              ed.属性.速度=Math.floor((ed.属性.速度||0)*spdMul);
              ed._越级速度已修正=true;
              needWrite=true;
            }
          }
          if(needWrite)await Mvu.replaceMvuData(mvd,{type:'message',message_id:G.aiMsgId});
        }
      }
    }catch(e){console.warn('[越级压制] MVU写入失败',e);}
    // 同步内存actor（actor无标记则补乘）
    for(const ek of G.eKeys){
      const ac=gA(ek);
      if(ac&&!ac._overLvSpdDone){
        ac.attr.速度=Math.floor((ac.attr.速度||0)*spdMul);
        ac._overLvSpdDone=true;
      }
    }
    if(G._overLvTier===1)log('实力差距过大，敌人气场极强，我方受到压制。','l-sys');
    if(G._overLvTier===2)log('实力差距极大，敌人气场如山岳一般，我方被压的喘不过气。','l-sys');
  }
}

// ── 构建角色 ───────────────────────────────────────────────
function buildActor(key,type,data){
  const info=data.基础信息||{}, core=data.核心状态||{}, attr=data.属性||{};
  const eq={力量:0,速度:0,防御力:0,内功修为:0,身法:0};
  if(type==='player'){
    const items=data.物品||{}, equip=data.装备||{};
    if(equip.主武器&&items.主武器类?.[equip.主武器]){
      const e=items.主武器类[equip.主武器].增幅效果||{};
      for(const k of Object.keys(eq))eq[k]+=(e[k]||0);
    }
    for(const slot of equip.饰品||[]){
      if(slot&&items.饰品类?.[slot]){
        const e=items.饰品类[slot].增幅效果||{};
        for(const k of Object.keys(eq))eq[k]+=(e[k]||0);
      }
    }
  }
  const mainKf=data.主运内功||'';
  const kfList=Array.isArray(data.领悟功法)?data.领悟功法:Object.keys(data.领悟功法||data.武功||{});
  if(type==='player'&&_customKfNames.length) for(const n of _customKfNames){if(!kfList.includes(n))kfList.push(n);}
  const bskObj=data.基本功||{};
  const _BK=['拳掌','棍枪','剑','刀','暗器','内功底蕴'];
  let _bSum=0; for(const k of _BK) _bSum+=(bskObj[k]||0);
  const hpBase=300+_bSum*8+(attr.力量||0)*20;
  const mpBase=300+_bSum*8+(attr.内功修为||0)*20;
  const hpMax=core.血量?.上限||hpBase;
  const mpMax=core.内力?.上限||mpBase;
  const weaponName=(data.装备||{}).主武器||'';
  const _pScore=calcCharPowerScore(attr,bskObj);
  return {
    key, type, name:info.姓名||key,
    title:info.称号||'',
    realm:info.境界||scoreToRealm(_pScore),
    powerScore:_pScore,
    hp:{cur:Math.min(core.血量?.当前??hpMax,hpMax),max:hpMax},
    mp:{cur:Math.min(core.内力?.当前??mpMax,mpMax),max:mpMax},
    attr:{...attr}, equipBonus:eq,
    bskill:bskObj,
    kungfu:kfList, mainKf, weaponName,
    items:data.物品||{},
    alive:true,
  };
}

function getAttr(key,name){
  const c=gA(key); if(!c)return 0;
  let v=(c.attr[name]||0)+(c.equipBonus?.[name]||0);
  for(const kfName of c.kungfu||[]){
    const kf=KUNGFU_LIB[kfName]; if(!kf)continue;
    const b=kf.基本功和属性增幅||{};
    v+=(b[name]||0);
  }
  const sts=G.statuses[key]||[];
  if(name==='力量'){
    const lUp=sts.find(s=>s.名称==='力量上升');
    if(lUp)v*=1+(lUp.数值||0.25);
  }
  if(name==='防御力'){
    const dUp=sts.find(s=>s.名称==='防御上升');
    if(dUp)v*=1+(dUp.数值||0.2);
    const dDn=sts.find(s=>s.名称==='防御下降');
    if(dDn)v*=1-(dDn.数值||0.2);
  }
  if(name==='身法'){
    const sUp=sts.find(s=>s.名称==='身法上升');
    if(sUp)v*=1+(sUp.数值||0.15);
  }
  if(name==='速度'){
    const spUp=sts.find(s=>s.名称==='速度上升');
    if(spUp)v*=1+(spUp.数值||0.1);
  }
  return Math.floor(v);
}

function getBskill(key,name){
  const c=gA(key); if(!c)return 0;
  let v=c.bskill?.[name]||0;
  for(const kfName of c.kungfu||[]){
    const kf=KUNGFU_LIB[kfName]; if(!kf)continue;
    const b=kf.基本功和属性增幅||{};
    v+=(b[name]||0);
  }
  return v;
}

// ── ATB循环 ─────────────────────────────────────────────
function atbLoop(){
  atbTimer=setInterval(()=>{
    if(G.phase!=='atb')return;
    // 先给所有人加进度（存到临时变量）
    const next={};
    let readyKey=null;
    for(const key of allKeys()){
      const c=gA(key); if(!c||!c.alive)continue;
      const spd=getAttr(key,'速度')||50;
      next[key]=(G.atbProg[key]||0)+(spd*0.05+1.5)*_speedMul;
    }
    // 找第一个到达100的
    for(const key of allKeys()){
      if(next[key]>=100&&!readyKey){
        readyKey=key; next[key]=100;
      }
    }
    // 如果有人到达，只更新到达者为100，其他人保持原值不变
    if(readyKey){
      G.atbProg[readyKey]=100;
    } else {
      // 没人到达，正常更新所有人
      for(const key of Object.keys(next)) G.atbProg[key]=next[key];
    }
    renderAtb();
    if(readyKey){
      G.phase='acting'; G.curKey=readyKey;
      onActorReady(readyKey);
    }
  },80);
}
function resumeAtb(){ G.phase='atb'; }

// ── 行动准备 ─────────────────────────────────────────────
function onActorReady(key){
  const c=gA(key); if(!c||!c.alive){resumeAtb();return;}
  // 高亮当前行动者
  document.querySelectorAll('.char-card').forEach(el=>el.classList.remove('acting-now'));
  q(`card-${key}`)?.classList.add('acting-now');
  G.roundCount++;
  renderAtb();
  // 清除该角色的防御/保护状态（只持续到下次行动）
  G.defending.delete(key);
  for(const[pk,gk]of Object.entries(G.jiazhuMap)){
    if(gk===key){ delete G.jiazhuMap[pk]; q(`card-${key}`)?.classList.remove('jiazhu'); }
  }
  applyPassiveHeal(key);
  if(!c.alive){resumeAtb();return;}

  const ctrl=(G.statuses[key]||[]).some(s=>['点穴','眩晕','冰冻'].includes(s.名称));
  if(ctrl){
    log(`${c.name} 受控制，无法行动`,'l-buff');
    endTurn(key); return;
  }
  const sealed=(G.statuses[key]||[]).some(s=>s.名称==='封穴');

  log(`─── ${c.name} 的回合 ───`,'l-sys');
  q('turn-banner').textContent=`[ ${c.name} 的回合 ]`;

  if(c.type==='player'||c.type==='ally'){
    if(_autoMode){
      G.phase='enemyTurn';
      q('action-zone').classList.remove('off');
      _syncActionBtns();
      q('actor-lbl').textContent=`— ${c.name} 自动操作 —`;
      setTimeout(()=>autoAct(key),spd(400));
    } else {
      G.phase='waitInput';
      q('action-zone').classList.remove('off');
      q('actor-lbl').textContent=`— ${c.name} 的操作 —`;
      clearAllPanels(); clearBtns();
      q('b-flee').style.display=c.type==='player'?'':'none';
      // 封穴时禁用武功按钮
      if(sealed){ q('b-kf').disabled=true; q('b-kf').style.opacity='0.3'; }
      else { q('b-kf').disabled=false; q('b-kf').style.opacity=''; }
    }
  } else {
    G.phase='enemyTurn';
    q('action-zone').classList.add('off');
    setTimeout(()=>enemyAct(key),spd(700));
  }
}

// ── 玩家操作入口 ──────────────────────────────────────────
function onAct(type){
  if(G.phase!=='waitInput'&&G.phase!=='selectTarget')return;
  // 如果在选目标阶段点了其他按钮，先取消当前选择
  if(G.phase==='selectTarget'){
    G.pending=null;
    exitSelect();
    G.phase='waitInput';
  }
  clearAllPanels(); clearBtns();

  if(type==='attack'){
    G.pending={type:'attack'};
    enterSelect('enemy');
    q('b-atk').classList.add('on');
  } else if(type==='kungfu'){
    q('b-kf').classList.add('on');
    buildKfPanel();
  } else if(type==='defend'){
    doDefend();
  } else if(type==='jiazhu'){
    const allies=aliveP().filter(k=>k!==G.curKey);
    if(!allies.length){
      toast('没有可保护的对象');
      return;
    }
    G.pending={type:'jiazhu'};
    q('b-jz').classList.add('on');
    q('tgt-hint').querySelector('span').textContent='▼ 选择要保护的队友 ▼';
    enterSelect('ally');
    q(`card-${G.curKey}`)?.classList.remove('selectable');
  } else if(type==='item'){
    q('b-item').classList.add('on');
    buildItemPanel();
  } else if(type==='flee'){
    doFlee();
  }
}

function onCardClick(key){
  if(G.phase!=='selectTarget')return;
  const c=gA(key); if(!c||!c.alive)return;
  const isE=G.eKeys.includes(key);
  if(G.tgtMode==='enemy'&&!isE)return;
  if(G.tgtMode==='ally'&&isE)return;
  if(G.tgtMode==='ally_all'&&isE)return;
  exitSelect();
  q(`card-${key}`)?.classList.add('targeted');
  G.phase='acting';
  const act=G.pending; if(!act)return;
  G.pending=null;

  if(act.type==='attack'){
    doNormalAttack(G.curKey,key);
  } else if(act.type==='jiazhu'){
    doJiazhu(G.curKey,key);
  } else if(act.type==='item'){
    doItem(key,act.itemName,act.itemRef);
  } else if(act.type==='kungfu'){
    doKungfu(G.curKey,act.kf,[key]);
  } else if(act.type==='kf_adj'){
    const all=aliveE(), idx=all.indexOf(key), tgts=[key];
    const candidates=[];
    if(idx>0)candidates.push(all[idx-1]);
    if(idx<all.length-1)candidates.push(all[idx+1]);
    if(candidates.length){
      const side=candidates[Math.floor(Math.random()*candidates.length)];
      tgts.push(side);
      log(`【相邻】同时波及 ${gA(side)?.name}`,'l-buff');
    }
    doKungfu(G.curKey,act.kf,tgts);
  }
}

// ── 武功武器类型判定 ──
function getKfWeaponType(kfName, kf){
  const req=kf?.基本功要求||{};
  let best='',bestV=0;
  for(const[k,v]of Object.entries(req)){if(v>bestV){bestV=v;best=k;}}
  if(!best||best==='内功底蕴')return 'palm';
  if(best==='拳掌') return /拳/.test(kfName)?'fist':'palm';
  if(best==='棍枪') return /枪/.test(kfName)?'spear':'staff';
  if(best==='剑')   return 'sword';
  if(best==='刀')   return 'blade';
  if(best==='暗器') return 'dart';
  return 'palm';
}

// 基本功名→免伤状态名映射
const BSKILL_TO_IMM={'拳掌':'免伤拳掌','棍枪':'免伤棍枪','剑':'免伤剑','刀':'免伤刀','暗器':'免伤暗器'};
// 武器类型→基本功名映射（用于免伤判定）
const WTYPE_TO_BSKILL={fist:'拳掌',palm:'拳掌',sword:'剑',blade:'刀',dart:'暗器',spear:'棍枪',staff:'棍枪'};
const BSKILL_TO_WTYPE={拳掌:'fist',剑:'sword',刀:'blade',暗器:'dart',棍枪:'spear'};

// 获取攻击者普攻的攻击类型（基本功最高的那个）
function getNormalAtkBskill(atkKey){
  const SKILLS=['拳掌','棍枪','剑','刀','暗器'];
  let best='拳掌',bestV=0;
  for(const sk of SKILLS){const v=getBskill(atkKey,sk);if(v>bestV){bestV=v;best=sk;}}
  return best;
}

// 检查目标是否有匹配的免伤状态，返回匹配的状态或null
// bskillName: 基本功名（拳掌/剑/刀/暗器/棍枪/内功底蕴）
function checkImmunity(defKey, bskillName){
  const sts=G.statuses[defKey]||[];
  // 全类型免伤
  const full=sts.find(s=>s.名称==='免伤');
  if(full) return full;
  // 分类免伤
  const immName=BSKILL_TO_IMM[bskillName];
  if(immName){const sp=sts.find(s=>s.名称===immName);if(sp)return sp;}
  return null;
}

// 根据武器名推断对应基本功
function getWeaponSkill(key){
  const c=gA(key);
  if(!c?.weaponName) return getNormalAtkBskill(key);
  // 从武器佩戴要求推断武器类型
  const items=c.items||c.playerItemsRef||{};
  const wData=items.主武器类?.[c.weaponName];
  if(wData?.佩戴要求){
    const req=wData.佩戴要求;
    const SKILLS=['拳掌','棍枪','剑','刀','暗器'];
    let best='',bestV=0;
    for(const sk of SKILLS){if((req[sk]||0)>bestV){bestV=req[sk];best=sk;}}
    if(best)return best;
  }
  return getNormalAtkBskill(key);
}

// ── 普攻 ─────────────────────────────────────────────────
function doNormalAttack(atkKey,tgtKey){
  const atk=gA(atkKey),tgt=gA(tgtKey); if(!atk||!tgt){endTurn(atkKey);return;}
  const aShn=getAttr(atkKey,'身法'), tShn=getAttr(tgtKey,'身法');
  let evade=clamp(5+(tShn-aShn)/100*10,5,70)/100;
  const evUp=(G.statuses[tgtKey]||[]).find(s=>s.名称==='闪避提升');
  if(evUp) evade=Math.min(0.7, evade+(evUp.数值||0.2));

  let def=getAttr(tgtKey,'防御力'); const _tgtDef=G.defending.has(tgtKey); if(_tgtDef)def*=2;
  dmg=dmg*(1-def/(def+200));
  if(G._overLvTier>=2&&G.eKeys.includes(atkKey))dmg*=1.7;
  // 攻击上升/下降：最终伤害乘数
  const _atkUp=(G.statuses[atkKey]||[]).find(s=>s.名称==='攻击上升');
  if(_atkUp)dmg*=1+(_atkUp.数值||0.2);
  const _atkDn=(G.statuses[atkKey]||[]).find(s=>s.名称==='攻击下降');
  if(_atkDn)dmg*=1-(_atkDn.数值||0.2);
  const _xhmg=(G.statuses[atkKey]||[]).find(s=>s.名称==='血海魔功');
  if(_xhmg)dmg*=1+(_xhmg.数值||0.5);
  dmg=Math.floor(dmg);
  if(_tgtDef)playSfx(SFX.defend); else {const _wt=BSKILL_TO_WTYPE[wSkill]||'fist';playSfx(getAtkSfx(_wt));}
  const reb=(G.statuses[tgtKey]||[]).find(s=>s.名称==='反弹');
  if(reb){const rv=Math.floor(dmg*(reb.数值||0.3));dealDmg(atkKey,rv);log(`${tgt.name}反弹 ${rv} 伤害`,'l-dmg');floatDmg(q(`card-${atkKey}`),`反弹-${rv}`,'hsl(280,60%,60%)');}

  const drain=(G.statuses[atkKey]||[]).find(s=>s.名称==='吸取内力');
  if(drain&&gA(tgtKey)?.alive){const dv=Math.floor(dmg*(drain.数值||0.1));const tc=gA(tgtKey);const ac=gA(atkKey);const stolen=Math.min(dv,tc.mp.cur);tc.mp.cur-=stolen;ac.mp.cur=Math.min(ac.mp.max,ac.mp.cur+stolen);updateCard(tgtKey);updateCard(atkKey);if(stolen>0)log(`${atk.name} 吸取 ${tgt.name} ${stolen} 内力`,'l-buff');}
  dealDmg(tgtKey,dmg);
  G.battleLog.push({t:'ATK',a:atk.name,v:tgt.name,d:dmg});
  const _st2=getAtkStyle(atkKey);
  log(`${atk.name}${_rF(_st2.verbs)}${tgt.name} -${dmg}`,'l-dmg');
  floatDmg(q(`card-${tgtKey}`),`-${dmg}`,'hsl(355,65%,60%)');
  // 反击
  const ctr=(G.statuses[tgtKey]||[]).find(s=>s.名称==='反击');
  if(ctr&&gA(tgtKey)?.alive&&gA(atkKey)?.alive){
    const cStr=getAttr(tgtKey,'力量'),cBval=getBskill(tgtKey,getWeaponSkill(tgtKey));
    let cDmg=Math.floor((cStr*1.5+cBval*1.5)*rand(0.9,1.1)*(ctr.数值||0.5));
    let cDef=getAttr(atkKey,'防御力');cDmg=Math.floor(cDmg*(1-cDef/(cDef+200)));
    dealDmg(atkKey,cDmg);log(`${tgt.name} 反击 → ${atk.name} -${cDmg}`,'l-dmg');
    floatDmg(q(`card-${atkKey}`),`-${cDmg}`,'hsl(30,70%,55%)');
  }
  endTurn(atkKey);
}

function selectKf(key,kfName,kf){
  clearAllPanels(); clearBtns();
  const range=kf.作用范围||'敌方单体';
  if(range.includes('仅我')){ G.phase='acting'; doKungfu(key,kfName,[key]); }
  else if(range.includes('我方')&&range.includes('群体')){ G.phase='acting'; doKungfu(key,kfName,aliveP()); }
  else if(range.includes('我方')||range.includes('友')){ G.pending={type:'kungfu',kf:kfName}; enterSelect('ally'); }
  else if(range.includes('群体')){ G.phase='acting'; doKungfu(key,kfName,aliveE()); }
  else if(range.includes('相邻')){ G.pending={type:'kf_adj',kf:kfName}; enterSelect('enemy'); }
  else { G.pending={type:'kungfu',kf:kfName}; enterSelect('enemy'); }
}

function doKungfu(atkKey,kfName,tgtKeys){
  const kf=KUNGFU_LIB[kfName]; if(!kf){endTurn(atkKey);return;}
  const atk=gA(atkKey); if(!atk){endTurn(atkKey);return;}
  if(atk.mp.cur<kf.消耗内力){log(`${atk.name} 内力不足！`,'l-miss');endTurn(atkKey);return;}
  atk.mp.cur=Math.max(0,atk.mp.cur-kf.消耗内力); updateCard(atkKey);
  // ── 小无相功特殊处理 ──
  if(kfName==='小无相功'){
    if((kf.冷却?.[0]||0)>0){ G.cds[atkKey]=G.cds[atkKey]||{}; G.cds[atkKey][kfName]=kf.冷却[0]+1; }
    const tgtKey=tgtKeys[0]; const tgt=gA(tgtKey);
    if(!tgt||!tgt.alive){log(`${atk.name} 施展小无相功，但目标已倒下`,'l-miss');endTurn(atkKey);return;}
    const atkPow=atk.powerScore||0;
    const allKf=(tgt.kungfu||[]).filter(n=>{
      const k=KUNGFU_LIB[n]; if(!k)return false;
      if(k.类别==='内功'&&(k.影响状态?.影响回合??-1)===-1)return false;
      return true;
    });
    if(!allKf.length){log(`${atk.name} 施展小无相功，但无法复制${tgt.name}的武功`,'l-miss');endTurn(atkKey);return;}
    const isPlayer=atkKey==='player';
    if(isPlayer){
      showCopyKfPanel(atkKey,tgtKey,allKf,atkPow);
    } else {
      const copyable=allKf.filter(n=>_canCopyKf(n,atkPow));
      if(!copyable.length){log(`${atk.name} 施展小无相功，但无法复制${tgt.name}的武功`,'l-miss');endTurn(atkKey);return;}
      const pick=copyable[Math.floor(Math.random()*copyable.length)];
      finishCopyKf(atkKey,tgtKey,pick);
    }
    return;
  }
  // 龙象之力：群体攻击前预消耗，存临时乘数
  const _lxHealKf=kf.作用范围&&(kf.作用范围.includes('我方')||kf.作用范围.includes('仅我'));
  let _lxMul=1;
  if(!_lxHealKf){const _lxB=(G.statuses[atkKey]||[]).find(s=>s.名称==='龙象之力');if(_lxB){const stacks=Math.round(_lxB.数值/0.1);_lxMul=1+_lxB.数值;G.statuses[atkKey]=G.statuses[atkKey].filter(s=>s.名称!=='龙象之力');updateCard(atkKey);log(`${atk.name}【龙象之力】${stacks}层爆发！伤害×${_lxMul.toFixed(1)}`,'l-crit');floatDmg(q(`card-${atkKey}`),`龙象×${stacks}`,'hsl(35,80%,55%)','16px');}}
  G._lxMul=_lxMul;
  const combos=[];
  for(const tk of tgtKeys){ const t=gA(tk); if(!t||!t.alive)continue; combos.push(doKfOnTgt(atkKey,kfName,kf,tk)); }
  G._lxMul=1;
  if((kf.冷却?.[0]||0)>0){ G.cds[atkKey]=G.cds[atkKey]||{}; G.cds[atkKey][kfName]=kf.冷却[0]+1; }
  if(combos.length) Promise.all(combos).then(()=>endTurn(atkKey));
  else endTurn(atkKey);
}

// ── 小无相功：武功选择面板 ──
function _canCopyKf(kfName,atkPow){
  const kf=KUNGFU_LIB[kfName]; if(!kf)return false;
  const req=kf.基本功要求||{};
  let reqSum=0; for(const v of Object.values(req)) reqSum+=v;
  return atkPow>=reqSum*4;
}

      </div>
      <div class="kf-mp">内力<br>${kf.消耗内力}</div>
      <div class="kf-cd rdy">CD${kf.冷却?kf.冷却[0]:0}</div>
      <div class="kf-rng">${rng}</div>
      ${!lvOk?`<div style="position:absolute;inset:0;display:flex;align-items:center;justify-content:center;color:#e44;font-size:15px;font-weight:bold;text-shadow:0 0 6px rgba(0,0,0,0.8);pointer-events:none">实力不足</div>`:''}`;
    if(lvOk)row.addEventListener('click',()=>{ clearAllPanels(); finishCopyKf(atkKey,tgtKey,kfName); });
    panel.appendChild(row);
  }
}

function finishCopyKf(atkKey,tgtKey,kfName){
  const atk=gA(atkKey),tgt=gA(tgtKey);
  G.copiedKf=G.copiedKf||{};
  G.copiedKf[atkKey]=kfName;
  applyStatus(atkKey,{名称:'小无相功',数值:0,剩余回合:6,来源:'小无相功'});
  log(`${atk.name} 施展小无相功，复制了${tgt?.name||''}的【${kfName}】！`,'l-buff');
  floatDmg(q(`card-${atkKey}`),`复制${kfName}`,'hsl(280,60%,65%)','14px');
  updateCard(atkKey);
  endTurn(atkKey);
}

async function doKfOnTgt(atkKey,kfName,kf,tgtKey){
  const atk=gA(atkKey),tgt=gA(tgtKey); if(!atk||!tgt||!tgt.alive)return;
  // 对自己施法（轻功/自buff）跳过命中判定
  if(atkKey!==tgtKey){
    // 命中判定（武功独立命中率，不受身法影响）
    let hitRate=kf.命中率??1;
    // 失明降低命中率
    const blind=(G.statuses[atkKey]||[]).find(s=>s.名称==='失明');
    if(blind) hitRate*=(1-(blind.数值||0.3));
    // 闪避上升：目标身上有则命中率再×(1-数值)
    const _tgtEvUp=(G.statuses[tgtKey]||[]).find(s=>s.名称==='闪避上升');
    if(_tgtEvUp) hitRate*=(1-(_tgtEvUp.数值||0.55));
    if(Math.random()>hitRate){
      const _mFx=getKfWeaponType(kfName,kf);
      await playAttackFx(_mFx,q(`card-${atkKey}`),q(`card-${tgtKey}`),kf.类别==='绝学'?kfName:null,()=>playSfx(SFX.miss));
      log(fL(FLAVOR.kfMiss,{atk:atk.name,kf:kfName,tgt:tgt.name}),'l-miss');
      floatDmg(q(`card-${tgtKey}`),'未中','#888','12px');
      G.battleLog.push({t:'DODGE',a:atk.name,v:tgt.name,sk:kfName});
      return;
    }
  }
  const base=kf.基础效果值||0;
  const floatR=kf.伤害浮动||[0.9,1.1];
  // 属性附加影响：字符串，取角色对应属性值
  const attrName=kf.属性附加影响||'无';
  const attrVal=attrName!=='无'?getAttr(atkKey,attrName):0;
  const attrMul=1+attrVal/(attrVal+200);
  // 基本功加成：取要求最高的基本功 + 内功底蕴
  let bskVal=0;
  const reqEntries=Object.entries(kf.基本功要求||{}).filter(([,v])=>v>0);
  if(reqEntries.length){
    let maxReq=0,maxName='';
    for(const [bName,req] of reqEntries){if(bName!=='内功底蕴'&&req>maxReq){maxReq=req;maxName=bName;}}
    if(maxName) bskVal+=getBskill(atkKey,maxName);
    bskVal+=getBskill(atkKey,'内功底蕴');
  }
  const bskMul=1+bskVal/150;
  let dmg=base*rand(floatR[0],floatR[1])*attrMul*bskMul;
  // 需要主运内功检查：未满足条件则威力大打折扣
  let _kfNerfed=false;
  if(kf.需要主运内功&&atk.mainKf!==kf.需要主运内功){
    dmg*=0.1; _kfNerfed=true;
    log(`${atk.name}未主运【${kf.需要主运内功}】，「${kfName}」威力大打折扣！`,'l-miss');
  }
  // 龙象之力：使用doKungfu预消耗的乘数
  if((G._lxMul||1)>1)dmg*=G._lxMul;
  // 暴击
  const critR=clamp(5+(getAttr(atkKey,'速度')-80)/10,5,15)/100;
  let critted=false;
  if(Math.random()<critR){ dmg*=1.5; critted=true; }
  // 防御减伤
  if(dmg>0){
    let def=getAttr(tgtKey,'防御力'); if(G.defending.has(tgtKey))def*=2;
    dmg=dmg*(1-def/(def+200));
  }
  if(G._overLvTier>=2&&G.eKeys.includes(atkKey))dmg*=1.7;
  // 攻击上升/下降：最终伤害乘数（治疗类不受影响）
  const isHealKf=kf.作用范围&&(kf.作用范围.includes('我方')||kf.作用范围.includes('仅我'));
  if(!isHealKf){
    const _atkUp2=(G.statuses[atkKey]||[]).find(s=>s.名称==='攻击上升');
    if(_atkUp2)dmg*=1+(_atkUp2.数值||0.2);
    const _atkDn2=(G.statuses[atkKey]||[]).find(s=>s.名称==='攻击下降');
    if(_atkDn2)dmg*=1-(_atkDn2.数值||0.2);
    const _xhmg2=(G.statuses[atkKey]||[]).find(s=>s.名称==='血海魔功');
    if(_xhmg2)dmg*=1+(_xhmg2.数值||0.5);
  }
  dmg=Math.floor(Math.max(0,dmg));
  // 反弹（基于实际承受伤害）
  const reb=(G.statuses[tgtKey]||[]).find(s=>s.名称==='反弹');
  if(reb&&dmg>0){const rv=Math.floor(dmg*(reb.数值||0.3));dealDmg(atkKey,rv);log(`${tgt.name}反弹 ${rv}`,'l-dmg');floatDmg(q(`card-${atkKey}`),`反弹-${rv}`,'hsl(280,60%,60%)');}
  // 治疗类（作用范围含"我方"或"仅我"）
  const isHeal=isHealKf;
  // 免伤判定（非治疗类武功）
  if(!isHeal&&dmg>0){
    const fxType=getKfWeaponType(kfName,kf);
    const kfBskill=WTYPE_TO_BSKILL[fxType]||'拳掌';
    const kfImm=checkImmunity(tgtKey,kfBskill);
    if(kfImm){
      // 仍然播放攻击特效
      await playAttackFx(fxType,q(`card-${atkKey}`),q(`card-${tgtKey}`),kf.类别==='绝学'?kfName:null);
      const tgtEl=q(`card-${tgtKey}`);
      const tgtR=tgtEl?tgtEl.getBoundingClientRect():{left:0,top:0,width:0,height:0};
      const immCx=tgtR.left+tgtR.width/2, immCy=tgtR.top+tgtR.height/2;
      log(`${atk.name}「${kfName}」击中${tgt.name}，被【${kfImm.名称}】抵消！`,'l-buff');
      floatDmg(tgtEl,'免疫','hsl(0,0%,95%)','15px');
      FxEngine.spawn(new FxRing({cx:immCx,cy:immCy,radius:50,color:[0,0,92],width:3,life:450}));
      FxEngine.spawn(new FxRing({cx:immCx,cy:immCy,radius:30,color:[0,0,85],width:2,life:350,delay:80}));
      G.battleLog.push({t:'IMM',a:atk.name,v:tgt.name,sk:kfName,st:kfImm.名称});
      return;
    }
  }
  // 攻击特效（非治疗类且有伤害才播放）
  if(!isHeal&&dmg>0){
    const fxType=getKfWeaponType(kfName,kf);
    const _hasAcu=kf.影响状态&&(kf.影响状态.名称==='点穴'||kf.影响状态.名称==='封穴');
    const _onHit=()=>{if(G.defending.has(tgtKey))playSfx(SFX.defend);else if(_hasAcu)playSfx(SFX.acupoint);else playSfx(getAtkSfx(fxType,kfName));};
    await playAttackFx(fxType,q(`card-${atkKey}`),q(`card-${tgtKey}`),kf.类别==='绝学'?kfName:null,_onHit);
  }
  if(isHeal&&dmg>0){
    playSfx(SFX.heal);
    tgt.hp.cur=Math.min(tgt.hp.max,tgt.hp.cur+dmg);
    updateCard(tgtKey);
    log(fL(FLAVOR.kfHeal,{atk:atk.name,kf:kfName,tgt:tgt.name})+` +${dmg}`,'l-heal');
    floatDmg(q(`card-${tgtKey}`),`+${dmg}`,'hsl(158,50%,55%)');
    fxHeal(q(`card-${tgtKey}`),'hp');
    G.battleLog.push({t:'HEAL',a:atk.name,v:tgt.name,sk:kfName,d:dmg});
  } else if(dmg>0){
    dealDmg(tgtKey,dmg);
    const color=critted?'var(--gold-bright)':'hsl(355,65%,60%)';
    const sz=critted?'22px':'17px';
    log(fL(critted?FLAVOR.kfCrit:FLAVOR.kfHit,{atk:atk.name,kf:kfName,tgt:tgt.name})+` -${dmg}${critted?' 暴击！':''}`,(critted?'l-crit':'l-dmg'));
    floatDmg(q(`card-${tgtKey}`),critted?`暴 -${dmg}`:`-${dmg}`,color,sz);
    G.battleLog.push({t:'KF',a:atk.name,v:tgt.name,sk:kfName,d:dmg,crit:critted});
  }
  // 无伤害的自施buff（如轻功），输出使用日志
  if(isHeal&&dmg===0){
    const effName=kf.影响状态?.名称||'';
    log(fL(FLAVOR.selfBuff,{atk:atk.name,kf:kfName,eff:effName||kf.描述||''}),'l-buff');
  }
  // 状态附加（名称为"无"时跳过；需要主运内功但未满足时跳过）
  if(kf.影响状态&&kf.影响状态.名称&&kf.影响状态.名称!=='无'&&tgt.alive&&!_kfNerfed){
    const s=kf.影响状态;
    // 即时生效（影响回合===0）：回血/回内直接结算，不挂持续状态
    if(s.影响回合===0&&(s.名称==='回血'||s.名称==='回内')){
      if(Math.random()<(s.成功率||0)){
        playSfx(SFX.heal);
        const val=s.数值||0;
        if(s.名称==='回血'){
          const h=val>0&&val<=1?Math.floor(tgt.hp.max*val):Math.floor(val);
          if(h>0){tgt.hp.cur=Math.min(tgt.hp.max,tgt.hp.cur+h);updateCard(tgtKey);
            log(`${atk.name}「${kfName}」回血 +${h}`,'l-heal');
            floatDmg(q(`card-${tgtKey}`),`+${h}`,'hsl(158,50%,55%)');fxHeal(q(`card-${tgtKey}`),'hp');
            G.battleLog.push({t:'HEAL',a:atk.name,v:tgt.name,sk:kfName,d:h});}
        } else {
          const m=val>0&&val<=1?Math.floor(tgt.mp.max*val):Math.floor(val);
          if(m>0){tgt.mp.cur=Math.min(tgt.mp.max,tgt.mp.cur+m);updateCard(tgtKey);
            log(`${atk.name}「${kfName}」回内 +${m}`,'l-heal');
            floatDmg(q(`card-${tgtKey}`),`+${m}`,'hsl(210,70%,55%)');fxHeal(q(`card-${tgtKey}`),'mp');
            G.battleLog.push({t:'HEAL',a:atk.name,v:tgt.name,sk:kfName,d:m});}
        }
      }
      return;
    }
    // 连击：逐次显示，0.5秒一击，miss也显示
    if(s.名称==='连击'){
      const hits=Math.max(1,Math.floor(s.数值||1));
      const comboSeq=[];
      for(let i=0;i<hits;i++){
        if(Math.random()<(s.成功率||0)) comboSeq.push(Math.floor(dmg*0.3*rand(0.8,1.2)));
        else comboSeq.push(0); // miss
      }
      return new Promise(resolve=>{
        floatDmg(q(`card-${tgtKey}`),'连 击','var(--gold-bright)','14px');
        log(`${atk.name}「${kfName}」触发连击 ×${hits}！`,'l-crit');
        let i=0,total=0;
        const tick=()=>{
          if(i>=hits||!gA(tgtKey)?.alive){
            if(total>0) G.battleLog.push({t:'KF',a:atk.name,v:tgt.name,sk:kfName+'·连击',d:total});
            resolve(); return;
          }
          const d=comboSeq[i];
          const _tgtEl=q(`card-${tgtKey}`);
          const _tr=_tgtEl?_tgtEl.getBoundingClientRect():null;
          const _atkEl=q(`card-${atkKey}`);
          if(_tr)playComboHitFx(kfName,kf,_atkEl,i,hits,_tr.left+_tr.width/2,_tr.top+_tr.height/2);
          const _fxType=getKfWeaponType(kfName,kf);
          playSfx(getAtkSfx(_fxType,kfName));
          if(d>0){
            dealDmg(tgtKey,d); total+=d;
            log(`${atk.name} 连击第${i+1}击 → ${tgt.name} -${d}`,'l-crit');
            floatDmg(q(`card-${tgtKey}`),`-${d}`,'var(--gold-bright)','15px');
          } else {
            log(`${atk.name} 连击第${i+1}击 → ${tgt.name} 未命中`,'l-miss');
            floatDmg(q(`card-${tgtKey}`),'未中','#888','12px');
          }
          i++;
          setTimeout(tick,spd(500));
        };
        setTimeout(tick,spd(500));
      });
    } else if(Math.random()<(s.成功率||0)){
      // 净化：清除目标负面状态
      if(s.名称==='净化'){
        const KEEP=['攻击上升','防御上升','回血','回内','吸取内力','反弹','反击','力量上升','身法上升','闪避提升','闪避上升','连击','免疫异常','速度上升','龙象之力'];
        const before=(G.statuses[tgtKey]||[]).length;
        G.statuses[tgtKey]=(G.statuses[tgtKey]||[]).filter(x=>KEEP.includes(x.名称)||/^免伤/.test(x.名称));
        const removed=before-(G.statuses[tgtKey]||[]).length;
        log(`${tgt.name} 被净化，清除 ${removed} 个负面状态`,'l-buff');
        updateCard(tgtKey);
      } else if(s.名称==='吸取内力'){
        // 即时吸取：从目标吸取内力给攻击者
        const ratio=s.数值||0.15;
        const stolen=Math.floor(tgt.mp.cur*ratio);
        if(stolen>0){
          tgt.mp.cur-=stolen;
          atk.mp.cur=Math.min(atk.mp.max,atk.mp.cur+stolen);
          updateCard(tgtKey);updateCard(atkKey);
          log(`${atk.name} 吸取 ${tgt.name} ${stolen} 内力`,'l-buff');
          floatDmg(q(`card-${tgtKey}`),`-${stolen}内力`,'hsl(280,50%,55%)','14px');
          floatDmg(q(`card-${atkKey}`),`+${stolen}内力`,'hsl(140,55%,55%)','14px');
          fxHeal(q(`card-${atkKey}`),'mp');
        }
      } else if(s.名称==='化去内力'){
        const ratio=s.数值||0.1;
        const drained=Math.floor(tgt.mp.cur*ratio);
        if(drained>0){
          tgt.mp.cur-=drained;
          updateCard(tgtKey);
          log(`${atk.name}「${kfName}」化去 ${tgt.name} ${drained} 内力`,'l-debuff');
          floatDmg(q(`card-${tgtKey}`),`-${drained}内力`,'hsl(280,50%,55%)','14px');
        }
      } else if(s.名称==='龙象之力'){
        // 龙象般若功：永久叠加层数，攻击时消耗
        G.statuses[atkKey]=G.statuses[atkKey]||[];
        const exist=G.statuses[atkKey].find(x=>x.名称==='龙象之力');
        if(exist){exist.数值+=s.数值||0.1;}
        else G.statuses[atkKey].push({名称:'龙象之力',数值:s.数值||0.1,剩余回合:-1});
        const stacks=Math.round((exist?exist.数值:(s.数值||0.1))/0.1);
        log(`${atk.name}【龙象般若功】蓄力，龙象之力 ×${stacks}`,'l-buff');
        floatDmg(q(`card-${atkKey}`),`龙象×${stacks}`,'hsl(35,80%,55%)','15px');
        updateCard(atkKey);
      } else if(s.名称==='吸血'){
        // 吸血：造成伤害的一定比例回复攻击者
        const ratio=s.数值||0.5;
        const heal=Math.floor(dmg*ratio);
        if(heal>0&&atk.alive){
          atk.hp.cur=Math.min(atk.hp.max,atk.hp.cur+heal);
          updateCard(atkKey);
          log(`${atk.name}「${kfName}」吸血 +${heal}`,'l-heal');
          floatDmg(q(`card-${atkKey}`),`+${heal}`,'hsl(0,60%,55%)','14px');
          fxHeal(q(`card-${atkKey}`),'hp');
        }
      } else {
        const isImm=/^免伤/.test(s.名称);
        const isImmAbn=s.名称==='免疫异常';
        // 好buff列表
        const GOOD_BUFFS=['攻击上升','防御上升','反弹','反击','力量上升','身法上升','闪避提升','闪避上升','速度上升','免疫异常'];
        const isGoodBuff=GOOD_BUFFS.includes(s.名称)||isImm;
        // 攻击技附带好buff（基础效果值非0 且 好buff）→ 给攻击者自己
        const isAtkWithGoodBuff=isGoodBuff&&(kf.基础效果值||0)!==0;
        // 免伤/免疫异常/攻击技好buff：挂给攻击者自己；其余：挂给目标
        const stKey=(isImm||isImmAbn||isAtkWithGoodBuff)?atkKey:tgtKey;
        const stTarget=(isImm||isImmAbn||isAtkWithGoodBuff)?atk:tgt;
        // 免疫异常拦截：目标身上有免疫异常时，debuff/控制无法施加
        const DEBUFFS=['点穴','眩晕','冰冻','封穴','流血','中毒','内伤','失明','攻击下降','防御下降','灼烧'];
        if(DEBUFFS.includes(s.名称)){
          const tgtSts=G.statuses[tgtKey]||[];
          if(tgtSts.find(x=>x.名称==='免疫异常')){
            const _el=q(`card-${tgtKey}`);
            const _r=_el?_el.getBoundingClientRect():{left:0,top:0,width:0,height:0};
            const _cx=_r.left+_r.width/2,_cy=_r.top+_r.height/2;
            log(`${tgt.name}【免疫异常】抵消了${s.名称}！`,'l-buff');
            floatDmg(_el,'免疫','hsl(0,0%,95%)','15px');
            FxEngine.spawn(new FxRing({cx:_cx,cy:_cy,radius:50,color:[0,0,92],width:3,life:450}));
            FxEngine.spawn(new FxRing({cx:_cx,cy:_cy,radius:30,color:[0,0,85],width:2,life:350,delay:80}));
            return;
          }
        }
        // 挂给自己的buff：+1补偿当回合扣减
        const _baseRounds=s.影响回合??3;
        const selfBuff=stKey===atkKey;
        const st={名称:s.名称,数值:s.数值||0,剩余回合:selfBuff?(_baseRounds+1):_baseRounds};
        G.statuses[stKey]=G.statuses[stKey]||[];
        const exist=G.statuses[stKey].find(x=>x.名称===st.名称);
        const NO_STACK=['点穴','眩晕','冰冻','反弹','反击','封穴'];
        const noStack=NO_STACK.includes(st.名称)||/^免伤/.test(st.名称);
        if(exist){exist.剩余回合=Math.max(exist.剩余回合,st.剩余回合);if(noStack)exist.数值=st.数值;else exist.数值+=st.数值;}
        else G.statuses[stKey].push(st);
        const _stBuff=['攻击上升','防御上升','回血','回内','反弹','反击','力量上升','身法上升','闪避提升','闪避上升','速度上升','免疫异常'].includes(st.名称)||/^免伤/.test(st.名称);
        if(_stBuff) log(fL(FLAVOR.statusBuff,{tgt:stTarget.name,st:st.名称,r:st.剩余回合}),'l-buff');
        else log(fL(FLAVOR.statusApply,{tgt:stTarget.name,st:st.名称,r:st.剩余回合}),'l-buff');
        floatDmg(q(`card-${stKey}`),st.名称,_stBuff?'hsl(42,75%,60%)':'hsl(0,70%,60%)','20px');
        G.battleLog.push({t:'STATUS',a:atk.name,v:stTarget.name,st:st.名称});
        updateCard(stKey);
      }
    }
  }
}

// ── 防御 ─────────────────────────────────────────────────
function doDefend(){
  const key=G.curKey,c=gA(key); if(!c){endTurn(key);return;}
  G.defending.add(key);
  log(`${c.name} 摆出防御架势，防御力翻倍`,'l-buff');
  G.battleLog.push({t:'DEF',a:c.name});
  endTurn(key);
}

// ── 招架面板 ─────────────────────────────────────────────
function buildJzPanel(){
  const panel=q('p-jz'); panel.innerHTML=''; panel.classList.add('open');

  item.数量=Math.max(0,(item.数量||1)-1); updateCard(tgtKey); endTurn(G.curKey);

// ── 逃跑 ─────────────────────────────────────────────────
function doFlee(){
  const key=G.curKey,c=gA(key); if(!c){endTurn(key);return;}
  const enemies=aliveE(); if(!enemies.length){endTurn(key);return;}
  let el=enemies[0],maxPow=gA(el)?.powerScore||0;
  for(const ek of enemies){const pw=gA(ek)?.powerScore||0;if(pw>maxPow){maxPow=pw;el=ek;}}
  const powD=Math.max(0,maxPow-(c.powerScore||0));
  const spdD=Math.max(0,(getAttr(key,'速度')-getAttr(el,'速度'))/200);
  const rate=clamp(0.9-powD*0.002+spdD,0.1,1.0);
  if(Math.random()<rate){log(`${c.name} 施展轻功，成功逃脱！`,'l-flee');endBattle('flee');}
  else{log(`${c.name} 逃跑被截住！`,'l-miss');endTurn(key);}
}

// ── 敌人AI ───────────────────────────────────────────────
function autoAct(key){
  const c=gA(key); if(!c||!c.alive){endTurn(key);return;}
  const es=aliveE(); if(!es.length){endTurn(key);return;}
  const tgtKey=es[Math.floor(Math.random()*es.length)];
  const sealed=(G.statuses[key]||[]).some(s=>s.名称==='封穴');
  let usedKf=null;
  if(!sealed){
    const hpPct=c.hp.cur/c.hp.max;
    const avail=[];
    for(const kfName of c.kungfu||[]){
      const kf=KUNGFU_LIB[kfName]; if(!kf||kf.消耗内力===0)continue;
      if(kf.类别==='轻功')continue;
      const cd=G.cds[key]?.[kfName]||0;
      if(cd>0||c.mp.cur<kf.消耗内力)continue;
      avail.push({name:kfName,kf});
    }
    // 血量<30%时优先回血
    if(hpPct<0.1){
      const heal=avail.find(a=>a.kf.影响状态&&a.kf.影响状态.名称==='回血');
      if(heal) usedKf=heal.name;
    }
    // 否则选基础效果值最高的
    if(!usedKf&&avail.length){
      const best=avail.reduce((a,b)=>(b.kf.基础效果值||0)>(a.kf.基础效果值||0)?b:a);
      usedKf=best.name;
    }
    // 检查复制武功
    if(!usedKf){
      const cpName=G.copiedKf?.[key];
      const cpSt=(G.statuses[key]||[]).some(s=>s.名称==='小无相功'&&s.剩余回合>0);
      if(cpName&&cpSt){
        const cpKf=KUNGFU_LIB[cpName];
        if(cpKf&&cpKf.消耗内力>0&&(G.cds[key]?.[cpName]||0)<=0&&c.mp.cur>=cpKf.消耗内力) usedKf=cpName;
      }
    }
  }
  if(usedKf){
    const kf=KUNGFU_LIB[usedKf];
    c.mp.cur=Math.max(0,c.mp.cur-kf.消耗内力); updateCard(key);
    const range=kf.作用范围||'';
    const isAlly=range.includes('我方')||range.includes('仅我');
    const pool=isAlly?aliveP():aliveE();
    let tgts;
    if(range.includes('仅我')){tgts=[key];}
    else if(range.includes('全体')||range.includes('群体')){tgts=[...pool];}
    else if(range.includes('相邻')){
      const pick=isAlly?pool[Math.floor(Math.random()*pool.length)]:tgtKey;
      tgts=[pick];
      if(pool.length>1){const idx=pool.indexOf(pick);const cd=[];if(idx>0)cd.push(pool[idx-1]);if(idx<pool.length-1)cd.push(pool[idx+1]);if(cd.length)tgts.push(cd[Math.floor(Math.random()*cd.length)]);}
    } else {
      tgts=isAlly?[pool[Math.floor(Math.random()*pool.length)]]:[tgtKey];
    }
    const fxP=[];
    for(const t of tgts)fxP.push(doKfOnTgt(key,usedKf,kf,t));
    if((kf.冷却?.[0]||0)>0){G.cds[key]=G.cds[key]||{};G.cds[key][usedKf]=kf.冷却[0]+1;}
    Promise.all(fxP).then(()=>endTurn(key));
  } else {
    doNormalAttack(key,tgtKey);
  }
}
function _classifyRole(kungfuList){
  let atk=0,sup=0,ctrl=0;
  const CTRLS=['点穴','眩晕','冰冻','封穴'];
  for(const name of kungfuList){
    const kf=KUNGFU_LIB[name]; if(!kf)continue;
    if(kf.影响状态&&CTRLS.includes(kf.影响状态.名称)){ctrl++;}
    else if(kf.影响状态&&(kf.影响状态.名称==='回血'||kf.影响状态.名称==='回内')){sup++;}
    else{atk++;}
  }
  if(ctrl>=atk&&ctrl>=sup)return 'ctrl';
  if(sup>=atk&&sup>=ctrl)return 'sup';
  return 'atk';
}
function _isHeal(kf){return kf.影响状态&&(kf.影响状态.名称==='回血'||kf.影响状态.名称==='回内');}
function _isCtrl(kf){const C=['点穴','眩晕','冰冻','封穴'];return kf.影响状态&&C.includes(kf.影响状态.名称);}
function _isAoe(kf){const r=kf.作用范围||'';return r.includes('群体')||r.includes('全体');}
function _tgtHasCtrl(tKey){return(G.statuses[tKey]||[]).some(s=>['点穴','眩晕','冰冻','封穴'].includes(s.名称));}

function enemyAct(key){
  const c=gA(key); if(!c||!c.alive){endTurn(key);return;}
  const ps=aliveP(); if(!ps.length){endTurn(key);return;}
  const es=aliveE();
  const sealed=(G.statuses[key]||[]).some(s=>s.名称==='封穴');
  const role=_classifyRole(c.kungfu||[]);
  const hpPct=c.hp.cur/c.hp.max;

  // 目标选择：输出型70%集火最低血量，控制型优先未被控目标
  let tgtKey;
  if(role==='atk'&&Math.random()<0.7){
    tgtKey=ps.reduce((a,b)=>(gA(a).hp.cur/gA(a).hp.max)<=(gA(b).hp.cur/gA(b).hp.max)?a:b);
  } else if(role==='ctrl'){
    const unCtrl=ps.filter(k=>!_tgtHasCtrl(k));
    tgtKey=unCtrl.length?unCtrl[Math.floor(Math.random()*unCtrl.length)]:ps[Math.floor(Math.random()*ps.length)];
  } else {
    tgtKey=ps[Math.floor(Math.random()*ps.length)];
  }

  // 保护判定
  if(G.jiazhuMap[tgtKey]&&Math.random()<0.8){
    const g=G.jiazhuMap[tgtKey];
    if(gA(g)?.alive){
      log(`${gA(g).name} 挺身保护 ${gA(tgtKey)?.name}！`,'l-buff');
      floatDmg(q(`card-${tgtKey}`),'被保护','hsl(158,50%,55%)','12px');
      tgtKey=g;
    }
  }

  let usedKf=null;
  if(!sealed){
    // 收集可用武功
    const avail=[];
    for(const kfName of c.kungfu||[]){
      const kf=KUNGFU_LIB[kfName]; if(!kf||kf.消耗内力===0)continue;
      const cd=G.cds[key]?.[kfName]||0;
      if(cd>0||c.mp.cur<kf.消耗内力)continue;
      avail.push({name:kfName,kf});
    }
    // 复制武功也加入候选
    const cpName=G.copiedKf?.[key];
    const cpSt=(G.statuses[key]||[]).some(s=>s.名称==='小无相功'&&s.剩余回合>0);
    if(cpName&&cpSt){
      const cpKf=KUNGFU_LIB[cpName];
      if(cpKf&&cpKf.消耗内力>0&&(G.cds[key]?.[cpName]||0)<=0&&c.mp.cur>=cpKf.消耗内力)
        avail.push({name:cpName,kf:cpKf});
    }

    // 自保：自身血量<20%时40%概率优先自回血
    if(hpPct<0.2&&Math.random()<0.1){
      const heal=avail.find(a=>_isHeal(a.kf)&&(a.kf.作用范围||'').includes('仅我'));
      if(heal) usedKf=heal.name;
    }
    // 治疗队友：队友血量<20%时40%概率治疗
    if(!usedKf){
      const hurtAlly=es.filter(k=>k!==key&&gA(k)?.alive&&gA(k).hp.cur/gA(k).hp.max<0.2);
      if(hurtAlly.length&&Math.random()<0.1){
        const heal=avail.find(a=>_isHeal(a.kf)&&!(a.kf.作用范围||'').includes('仅我'));
        if(heal) usedKf=heal.name;
      }
    }
    // 策略选择
    if(!usedKf){
      for(const {name,kf} of avail){
        // 轻功：20%概率使用
        if(kf.类别==='轻功'){if(Math.random()>0.2)continue; usedKf=name; break;}
        // 回血类：血量>20%时仅5%概率
        if(_isHeal(kf)){if(hpPct>=0.2&&Math.random()>0.05)continue; usedKf=name; break;}
        // 控制类：目标已被控则跳过
        if(_isCtrl(kf)&&_tgtHasCtrl(tgtKey))continue;
        // 群攻：输出型且目标≥3时优先
        if(role==='atk'&&_isAoe(kf)&&ps.length>=3){usedKf=name; break;}
        // 默认使用
        usedKf=name; break;
      }
    }
  }

  if(usedKf){
    const kf=KUNGFU_LIB[usedKf];
    c.mp.cur=Math.max(0,c.mp.cur-kf.消耗内力); updateCard(key);
    const range=kf.作用范围||'';
    let tgts;
    const isAlly=range.includes('我方')||range.includes('仅我');
    const pool=isAlly?aliveE():aliveP();
    if(range.includes('仅我')){
      tgts=[key];
    } else if(range.includes('全体')||range.includes('群体')){
      tgts=[...pool];
    } else if(range.includes('相邻')){
      const pick=isAlly?pool[Math.floor(Math.random()*pool.length)]:tgtKey;
      tgts=[pick];
      if(pool.length>1){
        const idx=pool.indexOf(pick);
        const cd=[];if(idx>0)cd.push(pool[idx-1]);if(idx<pool.length-1)cd.push(pool[idx+1]);
        if(cd.length)tgts.push(cd[Math.floor(Math.random()*cd.length)]);
      }
    } else {
      if(isAlly){
        // 治疗类：优先治疗血量最低的队友
        if(_isHeal(kf)){
          const lowest=pool.reduce((a,b)=>(gA(a).hp.cur/gA(a).hp.max)<=(gA(b).hp.cur/gA(b).hp.max)?a:b);
          tgts=[lowest];
        } else {tgts=[pool[Math.floor(Math.random()*pool.length)]];}
      } else {tgts=[tgtKey];}
    }
    const fxPromises=[];
    for(const t of tgts)fxPromises.push(doKfOnTgt(key,usedKf,kf,t));
    if((kf.冷却?.[0]||0)>0){G.cds[key]=G.cds[key]||{};G.cds[key][usedKf]=kf.冷却[0]+1;}
    Promise.all(fxPromises).then(()=>endTurn(key));
  } else {
    doNormalAttack(key,tgtKey);
  }
}

// ── 伤害 / 状态 ───────────────────────────────────────────
function dealDmg(key,dmg){
  const c=gA(key); if(!c||!c.alive)return;
  c.hp.cur=Math.max(0,c.hp.cur-dmg); updateCard(key);
  if(c.hp.cur<=0){
    c.alive=false;
    G.battleLog.push({t:'KO',v:c.name,side:c.type==='enemy'?'E':'P'});
    log(`${c.name} 倒下了！`,'l-sys'); updateCard(key);
    if(c.type==='enemy'){
      grantKillProf(key);
      const _isXinmo=/^\[心魔\]/.test(c.name);
      const gold=Math.floor((realmToEqLv(c.realm||'初学乍练'))*20*rand(0.5,2))*(_isXinmo?3:1);
      G.goldGained=(G.goldGained||0)+gold;
      if(gold>0) log(`掉落 ${gold} 金钱`,'l-buff');
    }
  }
}

// ── 计算武功血量/内力上限加成 ─────────────────────────────
function calcKfMaxBonus(key){
  const ac=gA(key); if(!ac)return{hp:0,mp:0};
  let hp=0, mp=0;
  for(const kfName of ac.kungfu||[]){
    const kf=KUNGFU_LIB[kfName]; if(!kf)continue;
    const b=kf.基本功和属性增幅||{};
    hp+=(b.血量上限||0); mp+=(b.内力上限||0);
  }
  return{hp,mp};
}

function grantKillProf(enemyKey){
  const e=gA(enemyKey); if(!e)return;
  const alive=aliveP(); if(!alive.length)return;
  const pPow=gA(G.pKeys[0])?.powerScore||0;
  const ePow=e.powerScore||0;
  const d=ePow-pPow;
  const m=d>=400?3:d>=200?2:d<=-400?0.5:d<=-200?0.8:1;
  const boss=/boss/i.test(e.name)||/boss/i.test(e.title||'');
  const elite=/精英/.test(e.name);
  const baseProf=Math.max(1,Math.floor(5*m*(boss?3:elite?1.5:1)));
  const _isXinmo=/^\[心魔\]/.test(e.name);
  const profAmt=_isXinmo?baseProf*2:baseProf;
  G.profGained=(G.profGained||0)+profAmt;
  log(`击败 ${e.name}，装备功法熟练度+${profAmt}${_isXinmo?' (心魔加倍)':''}`,'l-sys');
  for(const k of alive){
    floatDmg(q(`card-${k}`),`熟练度+${profAmt}`,'var(--gold)','13px');
  }
}

function applyStatus(key,s){
  if(!s||!s.名称)return;
  G.statuses[key]=G.statuses[key]||[];
  const exist=G.statuses[key].find(x=>x.名称===s.名称);
  if(exist){exist.剩余回合=Math.max(exist.剩余回合,s.剩余回合);exist.数值=s.数值;}
  else G.statuses[key].push({...s});
  updateCard(key);
}

function applyPassiveHeal(key){
  const c=gA(key); if(!c||!c.alive||!c.mainKf)return;
  const kf=KUNGFU_LIB[c.mainKf]; if(!kf)return;
  const eff=kf.影响状态; if(!eff||!eff.名称||eff.名称==='无')return;
  const val=kf.基础效果值||0;
  let healed=false;
  if(eff.名称==='回血'){
    const h=val<1?Math.floor(c.hp.max*val):Math.floor(val);
    if(h>0){c.hp.cur=Math.min(c.hp.max,c.hp.cur+h);log(`${c.name} 内功回血 +${h}`,'l-heal');healed=true;}
  } else if(eff.名称==='回内'){
    const m=val<1?Math.floor(c.mp.max*val):Math.floor(val);
    if(m>0){c.mp.cur=Math.min(c.mp.max,c.mp.cur+m);log(`${c.name} 内功回内 +${m}`,'l-heal');healed=true;}
  }
  if(healed)updateCard(key);
}

function processStatusTick(key){
  const sts=G.statuses[key]||[]; if(!sts.length)return;
  const c=gA(key); if(!c||!c.alive)return;
  for(const s of sts){
    if(s.名称==='流血'||s.名称==='中毒'||s.名称==='灼烧'){
      const d=s.数值>0&&s.数值<1?Math.floor(c.hp.max*s.数值):Math.floor(s.数值||c.hp.max*0.05);
      dealDmg(key,d);
      log(`${c.name}【${s.名称}】-${d}`,'l-dmg');
    }
    if(s.名称==='内伤'){
      const d=s.数值>0&&s.数值<1?Math.floor(c.mp.max*s.数值):Math.floor(s.数值||c.mp.max*0.05);
      if(c.mp.cur>=d){ c.mp.cur-=d; log(`${c.name}【内伤】内力-${d}`,'l-dmg'); }
      else { const lack=d-c.mp.cur; c.mp.cur=0; dealDmg(key,lack); log(`${c.name}【内伤】内力不足，气血-${lack}`,'l-dmg'); }
      updateCard(key);
    }
    if(s.名称==='回血'){
      const h=s.数值>0&&s.数值<1?Math.floor(c.hp.max*s.数值):Math.floor(s.数值||c.hp.max*0.05);
      c.hp.cur=Math.min(c.hp.max,c.hp.cur+h); updateCard(key);
      log(`${c.name}【回血】+${h}`,'l-heal');
      floatDmg(q(`card-${key}`),`+${h}`,'hsl(158,50%,55%)',14);fxHeal(q(`card-${key}`),'hp');
    }
    if(s.名称==='回内'){
      const m=s.数值>0&&s.数值<1?Math.floor(c.mp.max*s.数值):Math.floor(s.数值||c.mp.max*0.05);
      c.mp.cur=Math.min(c.mp.max,c.mp.cur+m); updateCard(key);
      log(`${c.name}【回内】+${m}`,'l-heal');
      floatDmg(q(`card-${key}`),`+${m}`,'hsl(210,70%,55%)',14);fxHeal(q(`card-${key}`),'mp');
    }
    if(s.名称==='九阳护体'){
      const h=Math.floor(c.hp.max*0.15);
      const m=Math.floor(c.mp.max*0.10);
      if(h>0){c.hp.cur=Math.min(c.hp.max,c.hp.cur+h);log(`${c.name}【九阳护体】回血+${h}`,'l-heal');floatDmg(q(`card-${key}`),`+${h}`,'hsl(158,50%,55%)',14);fxHeal(q(`card-${key}`),'hp');}
      if(m>0){c.mp.cur=Math.min(c.mp.max,c.mp.cur+m);log(`${c.name}【九阳护体】回内+${m}`,'l-heal');setTimeout(()=>{floatDmg(q(`card-${key}`),`+${m}`,'hsl(210,70%,55%)',14);fxHeal(q(`card-${key}`),'mp');},spd(500));}
      updateCard(key);
    }
    if(s.名称==='血海魔功'){
      const d=Math.floor(c.hp.max*0.2);
      dealDmg(key,d);
      log(`${c.name}【血海魔功】气血翻涌 -${d}`,'l-dmg');
      floatDmg(q(`card-${key}`),`-${d}`,'hsl(0,70%,50%)','14px');
      updateCard(key);
    }
    if(s.剩余回合!==-1)s.剩余回合--;
  }
  G.statuses[key]=sts.filter(s=>s.剩余回合===-1||s.剩余回合>0);
  // 小无相功状态到期，清除复制武功
  if(G.copiedKf?.[key]&&!(G.statuses[key]||[]).some(s=>s.名称==='小无相功')){
    log(`${(gA(key)?.name)||''}的小无相功效果消散，【${G.copiedKf[key]}】已失效`,'l-buff');
    delete G.copiedKf[key];
  }
  updateCard(key);
}

// ── 结算写MVU ─────────────────────────────────────────────
async function doSettle(){
  if(G.result==='win'&&!G.settleChoice){toast('请选择处置方式');return;}
  const btn=q('btn-settle'); btn.disabled=true; btn.textContent='结算中…';
  if(!getMvu()){toast('MVU未就绪');btn.disabled=false;btn.textContent='结算战斗';return;}
  try{
    const Mvu=getMvu();
    // 重新定位最新有效消息ID（避免战斗期间消息变动导致旧ID越界）
    let settleId=G.aiMsgId;
    const lastId=getLastMsgId();
    if(settleId===null||settleId>lastId) settleId=lastId;
    if(settleId<0){throw new Error('无法定位消息');}
    const data=await Mvu.getMvuData({type:'message',message_id:settleId});
    if(!data?.stat_data)throw new Error('无法获取stat_data');
    const sd=data.stat_data, res=G.result;

    if(sd.玩家){
      const pk=G.pKeys[0], pc=gA(pk);
      if(pc){
        sd.玩家.核心状态.血量.上限=pc.hp.max;
        sd.玩家.核心状态.内力.上限=pc.mp.max;
        sd.玩家.核心状态.血量.当前=pc.alive?pc.hp.cur:1;
        sd.玩家.核心状态.内力.当前=pc.mp.cur;
        sd.玩家.核心状态.状态列表=[];
        if(pc.attr) sd.玩家.属性={...sd.玩家.属性,...pc.attr};
      }
    }
    // 队友
    if(sd.我方队友){
      for(const name of Object.keys(sd.我方队友)){
        const ac=gA(name); if(!ac)continue;
        sd.我方队友[name].核心状态.血量.上限=ac.hp.max;
        sd.我方队友[name].核心状态.内力.上限=ac.mp.max;
        sd.我方队友[name].核心状态.血量.当前=ac.alive?ac.hp.cur:1;
        sd.我方队友[name].核心状态.内力.当前=ac.mp.cur;
        sd.我方队友[name].核心状态.状态列表=[];
        if(ac.attr) sd.我方队友[name].属性={...sd.我方队友[name].属性,...ac.attr};
      }
    }
    // 战斗熟练度增长（写入装备功法）
    if(G.profGained>0&&sd.武功){
      const equipped=sd.玩家?.装备功法||sd.玩家?.领悟功法||[];
      for(const kfName of equipped){
        if(sd.武功[kfName]) sd.武功[kfName].熟练度=(sd.武功[kfName].熟练度||0)+G.profGained;
      }
    }
    // 物品消耗
    if(sd.玩家?.物品){
      const pItems=gA('player')?.items||{};
      for(const cat of Object.keys(sd.玩家.物品)){
        const catObj=sd.玩家.物品[cat];
        if(typeof catObj!=='object')continue;
        for(const iName of Object.keys(catObj)){
          if(pItems[cat]?.[iName]){
            catObj[iName].数量=pItems[cat][iName].数量;
            if(catObj[iName].数量<=0)delete catObj[iName];
          }
        }
      }
    }
    // 清除敌人
    if(sd.敌人) sd.敌人={};
    // 金钱掉落
    if(res==='win'&&G.goldGained>0&&sd.玩家){
      sd.玩家.金钱=(sd.玩家.金钱||0)+G.goldGained;
    }

    await Mvu.replaceMvuData(data,{type:'message',message_id:settleId});
    sendBattleRecord();
    toast('结算完成');
    btn.textContent='已结算';
  }catch(e){
    console.error('结算失败',e);
    toast('结算失败：'+e.message);
    btn.disabled=false; btn.textContent='结算战斗';
  }
}

// ── 战报构建 ─────────────────────────────────────────────
function getWoundLevel(c){
  if(!c.alive)return 'KO';
  const r=c.hp.cur/c.hp.max;
  if(r>=1)return '无伤';
  if(r>=0.6)return '轻伤';
  if(r>=0.1)return '重伤';
  return '濒死';
}

function buildBattleRecord(){
  const res=G.result;
  const resLabel={win:'胜',lose:'败',flee:'逃'}[res]||res;

  // 我方状态
  const pList=G.pKeys.map(k=>{
    const c=gA(k); if(!c)return null;
    const wp=c.weaponName||'徒手';
    return `${c.name}使用${wp}:${getWoundLevel(c)}`;
  }).filter(Boolean);

  // 敌方状态
  const eList=G.eKeys.map(k=>{
    const c=gA(k); if(!c)return null;
    return `${c.name}:${getWoundLevel(c)}`;
  }).filter(Boolean);

  // 关键事件提取
  const events=[];
  const seen=new Set();
  // KO事件（最高优先级）
  for(const e of G.battleLog.filter(e=>e.t==='KO')){
    const txt=`${e.v}倒地`;
    if(!seen.has(txt)){seen.add(txt);events.push({p:0,txt});}
  }
  // 暴击/重创
  for(const e of G.battleLog.filter(e=>e.t==='KF')){
    if(e.crit){
      const txt=`${e.a}${e.sk}暴击${e.v}`;
      if(!seen.has(txt)){seen.add(txt);events.push({p:1,txt});}
    }
    const tgt=gA(G.eKeys.find(k=>gA(k)?.name===e.v)||G.pKeys.find(k=>gA(k)?.name===e.v));
    if(tgt&&e.d>=tgt.hp.max*0.2){
      const txt=`${e.a}用${e.sk}重创${e.v}`;
      if(!seen.has(txt)){seen.add(txt);events.push({p:1,txt});}
    }
  }
  // 治疗（只记一次）
  const healSeen=new Set();
  for(const e of G.battleLog.filter(e=>e.t==='HEAL')){
    const k=`${e.a}${e.sk}`;
    if(!healSeen.has(k)){healSeen.add(k);events.push({p:3,txt:`${e.a}用${e.sk}治疗${e.v}`});}
  }
  // 按优先级排序，最多6条
  events.sort((a,b)=>a.p-b.p);
  const keyEvents=events.slice(0,6).map(e=>e.txt);

  return `BATTLE_RECORD:结果=${resLabel}|回合=${G.roundCount}|我方=[${pList.join(',')}]|敌方=[${eList.join(',')}]|关键=[${keyEvents.join(',')}]${G.settleChoice?'|处置='+G.settleChoice:''}`;
}

function sendBattleRecord(){
  const record=buildBattleRecord();
  try{
    const ta=window.parent?.document?.getElementById('send_textarea');
    if(ta){
      const nativeSetter=Object.getOwnPropertyDescriptor(window.parent.HTMLTextAreaElement.prototype,'value').set;
      nativeSetter.call(ta,record);
      ta.dispatchEvent(new Event('input',{bubbles:true}));
    }
  }catch(e){console.warn('写入输入框失败',e);}
}

function renderSt(key){
  const BUFFS=['攻击上升','防御上升','回血','回内','吸取内力','反弹','反击','力量上升','身法上升','闪避提升','连击','免伤','免疫异常','速度上升','小无相功','龙象之力'];
  const CTRLS=['点穴','眩晕','冰冻','封穴'];
  return(G.statuses[key]||[]).map(s=>{
    const b=BUFFS.includes(s.名称)||/^免伤/.test(s.名称);
    const ctrl=CTRLS.includes(s.名称);
    const cls=b?'s-buff':ctrl?'s-ctrl':'s-debuff';
    // 龙象之力：显示层数
    if(s.名称==='龙象之力'){const stacks=Math.round(s.数值/0.1);return `<span class="s-bd ${cls}" title="龙象之力 ×${stacks}（+${Math.round(s.数值*100)}%伤害）">龙象之力(${stacks})</span>`;}
    if(s.剩余回合===-1&&s.来源){
      const useLabel=s.名称!==s.来源&&s.名称!=='回血'&&s.名称!=='回内'&&s.名称!=='免疫异常';
      const label=useLabel?s.名称:s.来源;
      const valStr=s.数值?(s.数值<1?Math.round(s.数值*100)+'%':s.数值):'';
      const tip=`${s.名称}${valStr?' '+valStr:''}（永久）— ${s.来源}`;
      return `<span class="s-bd ${cls}" title="${tip}">${label}</span>`;
    }
    const r=s.剩余回合===-1?'∞':s.剩余回合;
    return `<span class="s-bd ${cls}">${s.名称}(${r})</span>`;
  }).join('');
}
