function _jhDeletePreset(origName){

}

function openJhPanel(){
  _ensureJhDom();
  if(_jhPanel.classList.contains('show')){closeJhPanel();return}
  const btn=document.querySelector('#wx-jh-open');
  const rect=btn?.getBoundingClientRect();
  const vh=window.innerHeight;
  const gap=4;
  _jhPanel.style.top='';_jhPanel.style.bottom='';
  if(rect){
    const spaceBelow=vh-rect.bottom-gap;
    const spaceAbove=rect.top-gap;
    if(spaceBelow>=300||spaceBelow>=spaceAbove){
      _jhPanel.style.top=`${rect.bottom+gap}px`;
      _jhPanel.style.maxHeight=`${spaceBelow}px`;
    }else{
      _jhPanel.style.bottom=`${vh-rect.top+gap}px`;
      _jhPanel.style.maxHeight=`${spaceAbove}px`;
    }
  }else{
    _jhPanel.style.top='80px';
    _jhPanel.style.maxHeight=`${vh-100}px`;
  }
  _jhRenderBody();
  _jhOverlay.classList.add('show');
  _jhPanel.classList.add('show');
}

function _jhFillInput(text){
  const docs=[];
  try{docs.push(window.top.document)}catch(e){}
  try{docs.push(window.parent.document)}catch(e){}
  for(const doc of docs){
    try{
      const ta=doc.getElementById('send_textarea');
      if(ta){ta.value=text;ta.dispatchEvent(new Event('input',{bubbles:true}));ta.dispatchEvent(new Event('change',{bubbles:true}));ta.focus();break}
    }catch(e){}
  }
}
function openJhEdit(ctx){
  _ensureJhDom();
  const titleEl=_jhEditOverlay.querySelector('#wx-jh-edit-title');
  const nameEl=_jhEditOverlay.querySelector('#wx-jh-edit-name');
  const contentEl=_jhEditOverlay.querySelector('#wx-jh-edit-content');
  if(ctx){
    titleEl.textContent='编辑快捷回复';
    nameEl.value=ctx.name||'';
    contentEl.value=ctx.content||'';
  }else{
    titleEl.textContent='新增快捷回复';
    nameEl.value='';contentEl.value='';
  }
  _jhEditOverlay._ctx=ctx;
  _jhEditOverlay.classList.add('show');
  nameEl.focus();
  const saveBtn=_jhEditOverlay.querySelector('#wx-jh-edit-save');
  const newSave=saveBtn.cloneNode(true);
  saveBtn.parentNode.replaceChild(newSave,saveBtn);
  newSave.addEventListener('click',()=>{
    const n=nameEl.value.trim(), c=contentEl.value.trim();
    if(!n){showToast('请输入按钮名');return}
    if(!c){showToast('请输入发送内容');return}
    const editCtx=_jhEditOverlay._ctx;
    if(editCtx&&editCtx.type==='preset'){
      const edited=_jhGetEdited();
      edited[editCtx.origName]={name:n,content:c};
      _jhSave(_JH_LS_EDITED,edited);
    }else if(editCtx&&editCtx.type==='custom'){
      const customs=_jhGetCustom();
      customs[editCtx.index]={name:n,content:c};
      _jhSave(_JH_LS_CUSTOM,customs);
    }else{
      const customs=_jhGetCustom();
      customs.push({name:n,content:c});
      _jhSave(_JH_LS_CUSTOM,customs);
    }
    closeJhEdit();
    _jhRenderBody();
    showToast(editCtx?'已保存':'已新增');
  });
}

function _dzSyncLiquid(){
  const hpPct=_dzState.hpMax>0?_dzState.localHp/_dzState.hpMax:0;
  const mpPct=_dzState.mpMax>0?_dzState.localMp/_dzState.mpMax:0;
  const VH=318;
  // 用rect的y和height控制液面：y=顶部空白，height=液体高度
  const hpH=Math.round(VH*hpPct), hpY=VH-hpH;
  const mpH=Math.round(VH*mpPct), mpY=VH-mpH;
  // 填充rect
  const hpFill=document.getElementById('dz-hp-fill');
  const mpFill=document.getElementById('dz-mp-fill');
  const hpGlow=document.getElementById('dz-hp-glow');
  const mpGlow=document.getElementById('dz-mp-glow');
  if(hpFill){hpFill.setAttribute('y',hpY);hpFill.setAttribute('height',hpH)}
  if(mpFill){mpFill.setAttribute('y',mpY);mpFill.setAttribute('height',mpH)}
  if(hpGlow){hpGlow.setAttribute('y',hpY);hpGlow.setAttribute('height',hpH)}
  if(mpGlow){mpGlow.setAttribute('y',mpY);mpGlow.setAttribute('height',mpH)}
  // 波浪组定位到液面顶部
  const hpWaves=document.getElementById('dz-hp-waves');
  const mpWaves=document.getElementById('dz-mp-waves');
  if(hpWaves) hpWaves.setAttribute('transform',`translate(0,${hpY})`);
  if(mpWaves) mpWaves.setAttribute('transform',`translate(0,${mpY})`);
  document.getElementById('dz-val-hp').textContent=`${_dzState.localHp}/${_dzState.hpMax}`;
  document.getElementById('dz-val-mp').textContent=`${_dzState.localMp}/${_dzState.mpMax}`;
}

function _dzUpdateBtns(){
  const btnHp=document.getElementById('dz-btn-hp');
  const btnMp=document.getElementById('dz-btn-mp');
  const status=document.getElementById('dz-status');
  const isDying=_dzState.localHp<=1;
  if(isDying){
    btnHp.disabled=true; btnMp.disabled=true;
    status.textContent='濒死无法运功'; status.className='dz-status dying';
    btnHp.textContent='回血'; btnMp.textContent='回内';
    btnHp.classList.remove('running'); btnMp.classList.remove('running');
    return;
  }
  status.classList.remove('dying');
  const noMp=_dzState.localMp<=0;
  const hpFull=_dzState.localHp>=_dzState.hpMax;
  const mpFull=_dzState.localMp>=_dzState.mpMax;
  if(_dzState.mode==='hp'){
    btnHp.textContent='停止'; btnHp.classList.add('running'); btnHp.disabled=false;
    btnMp.disabled=true; btnMp.textContent='回内'; btnMp.classList.remove('running');
    status.textContent='正在以内力恢复伤势...';
  } else if(_dzState.mode==='mp'){
    btnMp.textContent='停止'; btnMp.classList.add('running'); btnMp.disabled=false;
    btnHp.disabled=true; btnHp.textContent='回血'; btnHp.classList.remove('running');
    status.textContent='正在打坐恢复内力...';
  } else {
    btnHp.textContent='回血'; btnMp.textContent='回内';
    btnHp.classList.remove('running'); btnMp.classList.remove('running');
    btnHp.disabled=noMp||hpFull; btnMp.disabled=mpFull;
    status.textContent='\u3000';
  }
}
function _dzTick(){
  if(_dzState.mode==='hp'){
    const amt=Math.max(1,Math.floor(_dzState.hpMax*_DZ_PCT_PER_TICK/100));
    const realAmt=Math.min(amt, _dzState.hpMax-_dzState.localHp, _dzState.localMp);
    if(realAmt<=0){_dzStop();return}
    _dzState.localHp+=realAmt; _dzState.localMp-=realAmt;
    _dzState.hpDelta+=realAmt; _dzState.mpDelta-=realAmt;
    if(_dzState.localMp<=0||_dzState.localHp>=_dzState.hpMax) _dzStop();
  } else if(_dzState.mode==='mp'){
    const amt=Math.max(1,Math.floor(_dzState.mpMax*_DZ_PCT_PER_TICK/100));
    const realAmt=Math.min(amt, _dzState.mpMax-_dzState.localMp);
    if(realAmt<=0){_dzStop();return}
    _dzState.localMp+=realAmt; _dzState.mpDelta+=realAmt;
    if(_dzState.localMp>=_dzState.mpMax) _dzStop();
  }
  _dzSyncLiquid(); _dzUpdateBtns();
}

function _dzToggle(mode){
  if(_dzState.mode===mode){_dzStop();return}
  if(_dzState.mode) _dzStop();
  _dzState.mode=mode;
  _dzState.timer=setInterval(_dzTick, _DZ_TICK);
  _dzUpdateBtns();
}

function _dzStop(){
  if(_dzState.timer){clearInterval(_dzState.timer);_dzState.timer=null}
  _dzState.mode=null;
  _dzUpdateBtns();
}

function _dzSelectChar(charKey){
  if(_dzState.mode) _dzStop();
  _dzFlush().then(()=>{
    _dzState.charKey=charKey;
    _dzState.hpDelta=0; _dzState.mpDelta=0;
    const ch=_dzGetCharData(charKey);
    if(!ch) return;
    const hp=ch.核心状态?.血量||{当前:0,上限:0};
    const mp=ch.核心状态?.内力||{当前:0,上限:0};
    _dzState.localHp=hp.当前; _dzState.hpMax=hp.上限;
    _dzState.localMp=mp.当前; _dzState.mpMax=mp.上限;
    _dzRefreshFigure(); _dzUpdateBtns();
    document.querySelectorAll('.dz-char-btn').forEach(b=>{
      b.classList.toggle('active',b.dataset.key===charKey);
    });
  });
}

async function _dzFlush(){

  const hpD=_dzState.hpDelta, mpD=_dzState.mpDelta;

  try{
    await window.parent.WX_MEDITATE({charKey:_dzState.charKey,hpDelta:hpD,mpDelta:mpD});
  }catch(e){console.error('[打坐flush]',e)}
}

async function doXinmoChallenge() {

  if (localStorage.getItem(`wx_${chatId}_心魔_${gameDate}`)) { showToast('今日已挑战过心魔'); return; }

    const lastId = window.parent.getLastMessageId?.();
    const msgs = await window.parent.getChatMessages?.(`0-${lastId}`);
    let aiMsgId = lastId;
    if (msgs) { for (let i = msgs.length - 1; i >= 0; i--) { if (msgs[i].role === 'assistant') { aiMsgId = msgs[i].message_id; break; } } }
    const data = await Mvu.getMvuData({ type: 'message', message_id: aiMsgId });
    if (!data?.stat_data?.玩家) { showToast('无法读取玩家数据'); return; }
    const p = data.stat_data.玩家;
    const pName = p.基础信息?.姓名 || '无名侠客';
    const xinmoKey = `[心魔]${pName}`;
    // 镜像玩家数据到敌人节点
    if (!data.stat_data.敌人) data.stat_data.敌人 = {};
    data.stat_data.敌人[xinmoKey] = {
      基础信息: {
        姓名: xinmoKey,
        性别: p.基础信息?.性别 || '男',

// 将弹窗overlay定位到iframe中用户实际可见的区域
function _positionOverlayAtViewport(overlay) {
  try {
    const fe = window.frameElement;
    if (fe) {
      const rect = fe.getBoundingClientRect();
      const parentVH = window.parent.innerHeight;
      const visibleTop = Math.max(0, -rect.top);
      const visibleBottom = Math.min(fe.clientHeight, -rect.top + parentVH);
      const visibleH = visibleBottom - visibleTop;
      if (visibleH > 50) {
        overlay.style.position = 'absolute';
        overlay.style.top = visibleTop + 'px';
        overlay.style.left = '0';
        overlay.style.right = '0';
        overlay.style.bottom = 'auto';
        overlay.style.height = visibleH + 'px';
        return;
      }
    }
  } catch(e) {}
  // fallback：frameElement不可用时尝试滚动父页面到iframe位置
  try { window.frameElement?.scrollIntoView({behavior:'smooth',block:'center'}); } catch(e) {}
}

        won = true;
        reward = prize;

    const res = await callParent('WX_LOTTERY_DRAW', { won, isFirst, reward, prob });

    if (won && !isFirst) {

      const newPrize = _randomJueXue(reward);
      _setLyPrize(chatId, newPrize);
    } else if (!won) {
      _setLyProb(chatId, prob + 0.5);

          for (let i = 0; i < 30; i++) {

            particleContainer.appendChild(pt);
          }
        }
        scrollLabel.textContent = isFirst ? '初入梦境，高人赐法' : '天赐机缘！';
        scrollName.textContent = reward;

// ══════════════════════════════════════════════
//  正文提取 — 严格只取最后一个 <content>…</content>
//  （内部不嵌套另一个 <content>，排除思维链提示文本）
// ══════════════════════════════════════════════
function extractContent(raw) {
  if (!raw) return '';
  // 从后往前找最后一个 </content>，再往前找最近的 <content>
  const closeTag = '</content>';
  const ci = raw.toLowerCase().lastIndexOf(closeTag);
  if (ci < 0) return '';
  const before = raw.slice(0, ci);
  const oi = before.toLowerCase().lastIndexOf('<content>');
  if (oi < 0) return '';
  return raw.slice(oi + 9, ci)
    .replace(new RegExp('\u003cimage\u003e[\\s\\S]*?\u003c/image\u003e','gi'), '')
    .replace(/<!--[\s\S]*?-->/g, '')
    .trim();
}

// ══════════════════════════════════════════════
//  选项提取 — 从最后一个 <w3g>…</w3g> 按行提取
// ══════════════════════════════════════════════
function extractOptions(raw) {
  if (!raw) return [];
  const all = [...raw.matchAll(/<w3g>([\s\S]*?)<\/w3g>/gi)];
  if (!all.length) return [];
  return all[all.length - 1][1]
    .split('\n')
    .map(l => l.trim())
    .filter(l => /^[A-Za-z][\uff1a:]/.test(l))
    .map(l => l.replace(/\s+/g, ' ').trim())
    .filter(Boolean);
}

function formatInline(raw) {
  if (!raw) return '';
  const ph = [];
  let t = raw;
  t = t.replace(/\*\*(.+?)\*\*/g, (_, c) => { const i = ph.length; ph.push(`<strong>${c}</strong>`); return `__WX_PH_${i}__`; });
  t = t.replace(/\*(.+?)\*/g,     (_, c) => { const i = ph.length; ph.push(`<em>${c}</em>`);         return `__WX_PH_${i}__`; });
  ph.forEach((html, i) => { t = t.replace(new RegExp(`__WX_PH_${i}__`, 'g'), html); });
  return t;
}

function renderMd(text) {
  return text.split(/\n+/).map(s => s.trim()).filter(Boolean).map(p => {
    const h4 = p.match(/^#{4}(?!#)\s+(.*)/);
    if (h4) return `<p style="font-size:21px">${formatInline(h4[1])}</p>`;
    const h3 = p.match(/^#{3}(?!#)\s+(.*)/);
    if (h3) return `<p style="font-size:21px">${formatInline(h3[1])}</p>`;
    const h2 = p.match(/^#{2}(?!#)\s+(.*)/);
    if (h2) return `<p style="font-size:24px">${formatInline(h2[1])}</p>`;
    const h1 = p.match(/^#{1}(?!#)\s+(.*)/);
    if (h1) return `<p style="font-size:28px">${formatInline(h1[1])}</p>`;
    return `<p>${formatInline(p)}</p>`;
  }).join('');
}

function buildPanel(t, charKey, char, isPlayer) {
  if (t==='attr')   return buildAttrPanel(char);

}

// ── 装备更换对比弹窗 ──
function showSwapModal(slot, charKey, currentName, candidates, char) {
  const modal=document.getElementById('wx-modal');
  const titleEl=document.getElementById('wx-modal-title');
  const textEl=document.getElementById('wx-modal-text');
  const actEl=document.getElementById('wx-modal-actions');
  titleEl.textContent='更换装备';

  const wornMap=getWornMap();
  const items=_state.玩家?.物品||{};
  const currentData=slot==='weapon'?(items.主武器类||{})[currentName]:(items.饰品类||{})[currentName];
  const COMPARE_KEYS=['力量','速度','防御力','内功修为','身法','血量值','内力值'];

  function buildCompareHtml(curData, newData, newName) {
    const curBonus=curData?.增幅效果||{};
    const newBonus=newData?.增幅效果||{};
    let rows='';
    const allKeys=new Set([...Object.keys(curBonus),...Object.keys(newBonus),...COMPARE_KEYS]);
    allKeys.forEach(k=>{
      if(!COMPARE_KEYS.includes(k)&&!curBonus[k]&&!newBonus[k]) return;
      const cv=curBonus[k]||0, nv=newBonus[k]||0, diff=nv-cv;
      const diffCls=diff>0?'up':diff<0?'down':'same';
      const diffTxt=diff>0?`↑${diff}`:diff<0?`↓${Math.abs(diff)}`:'—';
      rows+=`<div class="wx-compare-row">
        <span class="wx-compare-key">${k}</span>
        <span class="wx-compare-val">${nv} <span class="wx-compare-diff ${diffCls}">${diffTxt}</span></span>
      </div>`;
    });
    return rows||'<div style="font-size:11px;color:var(--text-dim);text-align:center;padding:8px">无增幅属性</div>';
  }

  let selectedName=null, selectedData=null;

  function render() {
    const curRows=buildCompareHtml(currentData, currentData, currentName);
    const newRows=selectedName?buildCompareHtml(currentData, selectedData, selectedName)
      :'<div style="font-size:11px;color:var(--text-dim);text-align:center;padding:16px">← 请从下方选择装备</div>';

    textEl.innerHTML=`
      <div class="wx-compare-grid">
        <div class="wx-compare-col">
          <div class="wx-compare-col-title">当前：${currentName||'无'}</div>
          ${curRows}
        </div>
        <div class="wx-compare-col">
          <div class="wx-compare-col-title">${selectedName?'更换：'+selectedName:'待选择'}</div>
          ${newRows}
        </div>
      </div>
      <div style="font-size:11px;color:var(--text-dim);margin-bottom:6px">可选装备：</div>
      <div class="wx-compare-candidates">
        ${candidates.length?candidates.map(([n,d])=>{
          const bonus=Object.entries(d.增幅效果||{}).map(([k,v])=>`${k}${v>0?'+':''}${v}`).join(' ');
          return `<div class="wx-compare-cand ${selectedName===n?'selected':''}" data-cand="${n}">
            ${n}${wornMap[n]?`<span style="font-size:10px;color:hsl(152,45%,55%);margin-left:4px">(${wornMap[n]}装备中)</span>`:''}${bonus?`<div class="wx-compare-cand-sub">${bonus}</div>`:''}
          </div>`;
        }).join(''):'<div class="wx-empty">无可更换的装备</div>'}
      </div>`;

    // 绑定候选点击
    textEl.querySelectorAll('[data-cand]').forEach(el=>{
      el.addEventListener('click',()=>{
        selectedName=el.dataset.cand;
        selectedData=candidates.find(([n])=>n===selectedName)?.[1]||null;
        render();
      });
    });

    actEl.innerHTML='';
    if(selectedName){
      const confirmBtn=document.createElement('button');
      confirmBtn.className='wx-modal-btn confirm';
      confirmBtn.textContent='确认更换';
      confirmBtn.addEventListener('click',async()=>{
        confirmBtn.disabled=true;
        const res=await callParent('WX_EQUIP',{charKey,slot,itemName:selectedName});
        if(res?.ok){showToast(`已更换为「${selectedName}」`);await loadState();refreshCard(charKey);modal.classList.remove('show');}
        else{showToast(res?.msg||'更换失败');confirmBtn.disabled=false;}
      });
      actEl.appendChild(confirmBtn);
    }
    const cancelBtn=document.createElement('button');
    cancelBtn.className='wx-modal-btn';
    cancelBtn.textContent='取消';
    cancelBtn.addEventListener('click',()=>modal.classList.remove('show'));
    actEl.appendChild(cancelBtn);
  }

  render();
  modal.classList.add('show');
  modal.addEventListener('click',e=>{if(e.target===modal)modal.classList.remove('show');},{once:true});
}

function buildTargets() {
  const t=[{label:'玩家',key:'player'}];
  Object.entries(_state?.我方队友||{}).forEach(([k,c])=>t.push({label:c.基础信息?.姓名||k,key:k}));
  return t;
}

function initAttrTooltip(){
  let tip=document.getElementById('wx-attr-tip-global');
  if(!tip){tip=document.createElement('div');tip.id='wx-attr-tip-global';tip.className='wx-attr-tooltip-global';document.body.appendChild(tip);}
  let _hideTimer=null;
  function valCls(v){return v>0?'pos':v<0?'neg':'zero';}
  function fmtVal(v){return (v>0?'+':'')+v;}
  function buildTipHtml(d){
    if(d.isBsk) return `<div class="wx-attr-tip-title">${d.n} 详情</div>`
      +`<div class="wx-attr-tip-row"><span class="lbl">加点</span><span class="val base">${d.base}</span></div>`
      +`<div class="wx-attr-tip-row"><span class="lbl">武学加成</span><span class="val ${valCls(d.kf)}">${fmtVal(d.kf)}</span></div>`
      +`<div class="wx-attr-tip-total"><span class="lbl">总计</span><span class="val">${d.total}</span></div>`;
    return `<div class="wx-attr-tip-title">${d.n} 详情</div>`
      +`<div class="wx-attr-tip-row"><span class="lbl">基础属性</span><span class="val base">${d.base}</span></div>`
      +`<div class="wx-attr-tip-row"><span class="lbl">装备加成</span><span class="val ${valCls(d.eq)}">${fmtVal(d.eq)}</span></div>`
      +`<div class="wx-attr-tip-row"><span class="lbl">武学加成</span><span class="val ${valCls(d.kf)}">${fmtVal(d.kf)}</span></div>`
      +`<div class="wx-attr-tip-row"><span class="lbl">基本功加成</span><span class="val ${valCls(d.bf)}">${fmtVal(d.bf)}</span></div>`
      +`<div class="wx-attr-tip-total"><span class="lbl">总计</span><span class="val">${d.total}</span></div>`;
  }
  function showTip(anchor){
    clearTimeout(_hideTimer);
    try{ var d=JSON.parse(anchor.dataset.attrtip); }catch(e){return;}
    tip.innerHTML=buildTipHtml(d);
    tip.classList.add('show');
    const r=anchor.getBoundingClientRect();
    const tw=tip.offsetWidth, th=tip.offsetHeight;
    let left=r.left-tw-10, top=r.top+r.height/2-th/2;
    if(left<4) left=r.right+10;
    if(top<4) top=4;
    if(top+th>window.innerHeight-4) top=window.innerHeight-th-4;
    tip.style.left=left+'px'; tip.style.top=top+'px';
  }
  function hideTip(){ _hideTimer=setTimeout(()=>{ tip.classList.remove('show'); },120); }
  document.addEventListener('mouseover',e=>{
    const anchor=e.target.closest('[data-attrtip]');
    if(anchor){ clearTimeout(_hideTimer); showTip(anchor); }
  });
  document.addEventListener('mouseout',e=>{
    const anchor=e.target.closest('[data-attrtip]');
    if(!anchor) return;
    const rel=e.relatedTarget;
    if(rel && anchor.contains(rel)) return;
    hideTip();
  });
}

function loadContent() {
  let raw = '';
  try {
    const lastId = getLastMessageId();
    if (lastId >= 0) {
      let targetMsg = null;
      // 方法1：用 -1 直接取最新 assistant 消息
      try {
        const msgs = getChatMessages(-1, { role: 'assistant' });
        if (msgs && msgs.length > 0) targetMsg = msgs[0];
      } catch(e) {}
      // 方法2：从最新 ID 向前扫描
      if (!targetMsg) {
        for (let id = lastId; id >= Math.max(0, lastId - 20); id--) {
          try {
            const msgs = getChatMessages(id, { role: 'assistant' });
            if (msgs && msgs.length > 0) { targetMsg = msgs[0]; break; }
          } catch(e) {}
        }
      }
      if (targetMsg) raw = targetMsg.message || '';
    }
  } catch(e) {}
  renderStory(raw);
  renderOptions(raw);
  // 检测：有敌人数据更新但无战斗标签（只看</content>之后，排除思维链干扰）
  const _cIdx = raw ? raw.lastIndexOf('</content>') : -1;
  const _tail = _cIdx !== -1 ? raw.substring(_cIdx) : raw;
  const hasEnemy = /\{\s*"op"\s*:\s*"insert"\s*,\s*"path"\s*:\s*"\/敌人/.test(_tail);
  const hasZd = _tail.includes('\u003czd\u003e');
  _battleAlert = hasEnemy && !hasZd;
  if(_battleAlert) showToast('检测到敌人，但似乎未出现战斗ui，您可手动打开战斗界面', 5000);
  loadState().then(() => { renderWorldInfo(); renderParty(); cleanOldRecords(); bgmOnContentLoad(raw); });
}
