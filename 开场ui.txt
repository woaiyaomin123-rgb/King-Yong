(function initBGM() {
  const audio = document.getElementById('bgm-audio');
  const btn   = document.getElementById('bgm-toggle');
  let playing = true;

  function tryPlay() {
    audio.play().catch(() => {});
  }

  // 尝试自动播放
  tryPlay();

  // 浏览器可能阻止自动播放，首次用户交互时恢复
  function resumeOnInteraction() {
    if (playing && audio.paused) tryPlay();
    document.removeEventListener('click', resumeOnInteraction);
    document.removeEventListener('keydown', resumeOnInteraction);
  }
  document.addEventListener('click', resumeOnInteraction);
  document.addEventListener('keydown', resumeOnInteraction);

  btn.addEventListener('click', (e) => {
    e.stopPropagation();
    playing = !playing;
    if (playing) {
      tryPlay();
      btn.textContent = '音乐开';
      btn.classList.remove('muted');
    } else {
      audio.pause();
      btn.textContent = '音乐关';
      btn.classList.add('muted');
    }
  });
})();

// ══════════════════════════════════════════════════════
//  粒子画布（飘散的金色尘埃）
// ══════════════════════════════════════════════════════
(function initParticles() {
  const canvas = document.getElementById('particle-canvas');
  const ctx    = canvas.getContext('2d');
  let W, H, particles;

  function resize() {
    W = canvas.width  = window.innerWidth;
    H = canvas.height = window.innerHeight;
  }

  function createParticles() {
    particles = Array.from({ length: 55 }, () => ({
      x:    Math.random() * W,
      y:    Math.random() * H,
      r:    0.4 + Math.random() * 1.2,
      vx:   (Math.random() - 0.5) * 0.18,
      vy:   -0.08 - Math.random() * 0.18,
      a:    Math.random(),
      va:   (Math.random() - 0.5) * 0.004,
    }));
  }

  function draw() {
    ctx.clearRect(0, 0, W, H);
    for (const p of particles) {
      p.x  += p.vx;
      p.y  += p.vy;
      p.a  += p.va;
      if (p.a < 0) { p.a = 0; p.va = Math.abs(p.va); }
      if (p.a > 1) { p.a = 1; p.va = -Math.abs(p.va); }
      if (p.y < -10) { p.y = H + 10; p.x = Math.random() * W; }
      if (p.x < -10) { p.x = W + 10; }
      if (p.x > W + 10) { p.x = -10; }

      ctx.beginPath();
      ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
      ctx.fillStyle = `hsla(42, 65%, 60%, ${p.a * 0.55})`;
      ctx.fill();
    }
    requestAnimationFrame(draw);
  }

  resize();
  createParticles();
  draw();
  window.addEventListener('resize', () => { resize(); createParticles(); });
})();

        </div>

        <div class="stat-btns">
          <button class="stat-btn" id="minus-${key}" onclick="adjustStat('${key}',-1)" aria-label="${key}减少">−</button>

  }).join('');
}

function buildProgress() {
  const row = document.getElementById('progress-row');
  row.innerHTML = PAGES.map((_,i) => `<div class="prog-dot" id="dot-${i}" aria-hidden="true"></div>`).join('');

function updateProgress() {
  PAGES.forEach((_,i) => {
    const d = document.getElementById('dot-' + i);
    d.className = 'prog-dot' + (i === currentPage ? ' active' : i < currentPage ? ' done' : '');
  });
  document.getElementById('page-counter').textContent = `${currentPage+1} / ${PAGES.length}`;
  document.getElementById('btn-prev').disabled = currentPage === 0;

  const curId = PAGES[currentPage].id;
  const hideNext = curId === 'choose_path' || curId === 'transmigrate';
  const isLast = currentPage === PAGES.length - 1;
  document.getElementById('btn-next').style.display    = (isLast || hideNext) ? 'none' : '';
  document.getElementById('btn-confirm').style.display = isLast ? '' : 'none';
}

function buildPages() {
  const wrap = document.getElementById('page-wrap');
  wrap.innerHTML = '';
  PAGES.forEach((p, i) => {
    const div = document.createElement('div');
    div.className = 'page' + (i === 0 ? ' active' : '');
    div.id = 'page-' + p.id;
    div.setAttribute('role', 'region');
    div.setAttribute('aria-label', p.label);
    div.innerHTML = getPageHTML(p.id);
    wrap.appendChild(div);
  });
}

function navigate(dir) {
  if (PAGES[currentPage].id === 'player1') syncPlayer1();

  const oldPage = document.getElementById('page-' + PAGES[currentPage].id);
  oldPage.classList.remove('active');
  oldPage.classList.add('exit');

  const curId = PAGES[currentPage].id;

  if (dir === 1 && curId === 'choose_path') {
    if (state.path === 'transmigrate') {
      currentPage = PAGES.findIndex(p => p.id === 'transmigrate');
    } else {
      currentPage = PAGES.findIndex(p => p.id === 'player1');
    }
  } else if (dir === -1 && (curId === 'transmigrate' || curId === 'player1')) {
    currentPage = PAGES.findIndex(p => p.id === 'choose_path');
  } else {
    currentPage = Math.max(0, Math.min(PAGES.length - 1, currentPage + dir));
  }

  setTimeout(() => { oldPage.classList.remove('exit'); }, 600);

  const newPage = document.getElementById('page-' + PAGES[currentPage].id);
  // 强制重排以触发动画
  void newPage.offsetWidth;
  newPage.classList.add('active');

  updateProgress();

  if (PAGES[currentPage].id === 'player2') {
    document.getElementById('stats-panel').innerHTML = renderStats();
    document.getElementById('pts-left').textContent  = state.pointsLeft;
    updateStatButtons();
  }
}

function adjustStat(key, delta) {
  if (delta > 0 && state.pointsLeft <= 0) return;
  if (delta < 0 && state.stats[key] <= BASE_VAL) return;
  state.stats[key]  += delta;
  state.pointsLeft  -= delta;

  document.getElementById('val-' + key).textContent = state.stats[key];

  const ptsEl = document.getElementById('pts-left');
  ptsEl.textContent = state.pointsLeft;
  ptsEl.classList.remove('flash');
  void ptsEl.offsetWidth;
  ptsEl.classList.add('flash');

  const pct = ((state.stats[key] - BASE_VAL) / TOTAL_PTS) * 100;
  document.getElementById('bar-' + key).style.width = Math.max(pct, 0) + '%';
  updateStatButtons();
}

function updateStatButtons() {
  STAT_KEYS.forEach(key => {
    const plus  = document.getElementById('plus-'  + key);
    const minus = document.getElementById('minus-' + key);
    if (plus)  plus.disabled  = state.pointsLeft <= 0;
    if (minus) minus.disabled = state.stats[key] <= BASE_VAL;
  });
}

async function confirmSetup() {
  if (!validate()) return;

  const btn = document.getElementById('btn-confirm');
  btn.disabled = true;
  btn.innerHTML = `
    <svg style="width:16px;height:16px;stroke:currentColor;fill:none;stroke-width:2;flex-shrink:0;animation:spin 1s linear infinite" viewBox="0 0 24 24">
      <circle cx="12" cy="12" r="10" stroke-dasharray="32" stroke-dashoffset="10"/>
    </svg>
    正在踏入…`;

  try {
    const Mvu = window.parent?.Mvu || window.Mvu;
    if (!Mvu) throw new Error('MVU未加载');

    let data = await Mvu.getMvuData({ type: 'message', message_id: 0 });
    if (!data) data = { stat_data: {} };
    if (!data.stat_data) data.stat_data = {};

    const _ = window.parent?._ || window._;

    const kf = KUNGFU_LIST.find(k => k.id === state.kungfu);
    const statKey = kf ? kf.stat : '拳掌';

    _.set(data, 'stat_data.玩家.基础信息.性别', state.gender);
    _.set(data, 'stat_data.$剧情控制', state.openingId);

    STAT_KEYS.forEach(key => {
      _.set(data, `stat_data.玩家.基本功加点.基本功.${key}`, state.stats[key]);
    });
    _.set(data, 'stat_data.玩家.基本功加点.点数', state.pointsLeft);

    _.set(data, 'stat_data.玩家.领悟功法', [state.kungfu]);

    // ── 根据初始武功注入初始武器 ──
    {
      const WEAPON_MAP = {
        '拳掌': '破指套',
        '棍枪': '破木棍',
        '剑':   '破铁剑',
        '刀':   '破铁刀',
        '暗器': '破飞镖',
      };
      const weaponName = WEAPON_MAP[statKey] || '破指套';
      _.set(data, `stat_data.玩家.物品.主武器类.${weaponName}`, {
        佩戴要求: { [statKey]: 5 },
        增幅效果: { 力量: 2, 防御力: 1 },
      });
      _.set(data, 'stat_data.玩家.装备.主武器', weaponName);
      console.log(`[开场界面] 初始武器：${weaponName}（${statKey}>=5，力量+2，防御力+1）`);
    }

    {
      const 等级 = 1;

      // 武功增幅
      const learnedList = [state.kungfu];
      const kfAttrBonus = { 力量: 0, 速度: 0, 防御力: 0, 内功修为: 0, 身法: 0 };
      const kfBskBonus  = {};
      let 武功血量加成 = 0, 武功内力加成 = 0;
      for (const kfName of learnedList) {
        const kfEntity = _kungfuLibCache?.[kfName];
        if (!kfEntity?.基本功和属性增幅) continue;
        const b = kfEntity.基本功和属性增幅;
        for (const k of ['力量','速度','防御力','内功修为','身法']) { if (b[k]) kfAttrBonus[k] += b[k]; }
        for (const k of STAT_KEYS) { if (b[k]) kfBskBonus[k] = (kfBskBonus[k] || 0) + b[k]; }
        武功血量加成 += (b.血量上限 ?? 0);
        武功内力加成 += (b.内力上限 ?? 0);
      }

      // 武功基本功增幅写入
      for (const k of STAT_KEYS) {
        if (kfBskBonus[k]) {
          const p = `stat_data.玩家.基本功加点.基本功.${k}`;
          _.set(data, p, (_.get(data, p, 0) || 0) + kfBskBonus[k]);
        }
      }

      // 装备增幅
      const equipBonus = { 力量: 0, 速度: 0, 防御力: 0, 内功修为: 0, 身法: 0 };
      let 装备血量加成 = 0, 装备内力加成 = 0;
      const weaponData = _.get(data, `stat_data.玩家.物品.主武器类.${_.get(data, 'stat_data.玩家.装备.主武器')}`);
      if (weaponData?.增幅效果) {
        const e = weaponData.增幅效果;
        for (const k of ['力量','速度','防御力','内功修为','身法']) { if (e[k]) equipBonus[k] += e[k]; }
        装备血量加成 += (e.血量值 ?? 0);
        装备内力加成 += (e.内力值 ?? 0);
      }

      // 基本功 → 属性加成（与日常UI公式一致）
      const BMAP = {
        拳掌:     { 力量: 0.5, 防御力: 0.2 },
        棍枪:     { 力量: 0.3, 速度: 0.3 },
        剑:       { 速度: 0.4, 身法: 0.3 },
        刀:       { 力量: 0.4, 防御力: 0.3 },
        暗器:     { 速度: 0.4, 身法: 0.4 },
        内功底蕴: { 内功修为: 0.8, 防御力: 0.2 },
      };
      const attrBonus = { 力量: 0, 速度: 0, 防御力: 0, 内功修为: 0, 身法: 0 };
      for (const sk of STAT_KEYS) {
        const sv = (state.stats[sk] || 0) + (kfBskBonus[sk] || 0);
        const mp = BMAP[sk];
        if (!mp) continue;
        for (const at in mp) {
          attrBonus[at] += Math.floor(sv * mp[at]);
        }
      }

      const 力量     = 10 + attrBonus['力量']     + kfAttrBonus['力量']     + equipBonus['力量'];
      const 速度     = 10 + attrBonus['速度']     + kfAttrBonus['速度']     + equipBonus['速度'];
      const 防御力   = 10 + attrBonus['防御力']   + kfAttrBonus['防御力']   + equipBonus['防御力'];
      const 内功修为 = 10 + attrBonus['内功修为'] + kfAttrBonus['内功修为'] + equipBonus['内功修为'];
      const 身法     = 10 + attrBonus['身法']     + kfAttrBonus['身法']     + equipBonus['身法'];

      _.set(data, 'stat_data.玩家.属性.力量',     力量);
      _.set(data, 'stat_data.玩家.属性.速度',     速度);
      _.set(data, 'stat_data.玩家.属性.防御力',   防御力);
      _.set(data, 'stat_data.玩家.属性.内功修为', 内功修为);
      _.set(data, 'stat_data.玩家.属性.身法',     身法);

      const 血量上限 = 300 + 等级 * 60 + 力量 * 20 + 武功血量加成 + 装备血量加成;
      const 内力上限 = 300 + 等级 * 60 + 内功修为 * 20 + 武功内力加成 + 装备内力加成;

      _.set(data, 'stat_data.玩家.核心状态.血量.上限', 血量上限);
      _.set(data, 'stat_data.玩家.核心状态.血量.当前', 血量上限);
      _.set(data, 'stat_data.玩家.核心状态.内力.上限', 内力上限);
      _.set(data, 'stat_data.玩家.核心状态.内力.当前', 内力上限);

      function expToNext(lv) {
        if (lv <= 30)  return lv * 60;
        if (lv <= 50)  return lv * 90;
        if (lv <= 70)  return lv * 150;
        if (lv <= 99)  return lv * 210;
        if (lv <= 120) return lv * 450;
        return lv * 600;
      }

      _.set(data, 'stat_data.玩家.基础信息.等级', 等级);
      _.set(data, 'stat_data.玩家.基础信息.经验值', 0);
      _.set(data, 'stat_data.玩家.基础信息.下一级所需经验', expToNext(等级));

      console.log(`[开场界面] 初始核心状态：血量上限=${血量上限}，内力上限=${内力上限}`);
    }

    let allKungfuRecord;
    if (_kungfuLibCache && typeof _kungfuLibCache === 'object' && Object.keys(_kungfuLibCache).length > 5) {
      allKungfuRecord = Object.fromEntries(
        Object.values(_kungfuLibCache).map(k => [k.名字, k.类别 || '普通'])
      );
      console.log('[开场界面] 武功库来源：内存缓存，共', Object.keys(allKungfuRecord).length, '个');
    } else {
      const getAllKungfu = _findGetAllKungfu();
      if (typeof getAllKungfu === 'function') {
        const lib = getAllKungfu();
        _kungfuLibCache = lib;
        allKungfuRecord = Object.fromEntries(
          Object.values(lib).map(k => [k.名字, k.类别 || '普通'])
        );
        console.log('[开场界面] 武功库来源：WX_GET_ALL_KUNGFU，共', Object.keys(allKungfuRecord).length, '个');
      } else {
        try {
          const priorData = await Mvu.getMvuData({ type: 'message', message_id: 0 });
          const priorKungfu = priorData?.stat_data?.武功;
          if (priorKungfu && Object.keys(priorKungfu).length > 5) {
            allKungfuRecord = priorKungfu;
            console.log('[开场界面] 武功库来源：MVU预注入，共', Object.keys(allKungfuRecord).length, '个');
          } else {
            throw new Error('MVU中武功库不完整');
          }
        } catch(mvuErr) {
          console.warn('[开场界面] 武功库兜底（仅5个初始武功）:', mvuErr.message);
          allKungfuRecord = Object.fromEntries(KUNGFU_LIST.map(k => [k.id, '普通']));
        }
      }
    }
    _.set(data, 'stat_data.武功', allKungfuRecord);

    await Mvu.replaceMvuData(data, { type: 'message', message_id: 0 });

    // ── 清空世界书"勿动"条目内容（避免上一次魂穿残留影响新角色）──
    try {
      const _p = window.parent;
      const _getNames = _p.getCharWorldbookNames || window.getCharWorldbookNames;
      const _getWB    = _p.getWorldbook           || window.getWorldbook;
      const _repWB    = _p.replaceWorldbook        || window.replaceWorldbook;
      if (_getNames && _getWB && _repWB) {
        const _books = _getNames('current');
        if (_books?.primary) {
          const _entries = await _getWB(_books.primary);
          const _target = _entries.find(e => e.name === '勿动');
          if (_target) {
            _target.content = '';
            await _repWB(_books.primary, _entries, { render: 'debounced' });
          }
        }
      }
    } catch(e) { console.warn('[开场界面] 清空"勿动"跳过:', e.message); }

    fillInput();

    showSuccess();

    await writeWorldbook();

  } catch(e) {
    console.error('[开场界面] 错误:', e);
    btn.disabled = false;
    btn.innerHTML = `
      <svg style="width:16px;height:16px;stroke:currentColor;fill:none;stroke-width:1.5;flex-shrink:0;" viewBox="0 0 24 24">
        <path d="M5 12h14M12 5l7 7-7 7"/>
      </svg>
      踏入江湖`;
    alert('发生错误：' + e.message);
  }
}

async function writeWorldbook() {
  try {
    const parent = window.parent;
    if (!parent) return;

    const getNames     = parent.getCharWorldbookNames  || window.getCharWorldbookNames;
    const createEntries = parent.createWorldbookEntries || window.createWorldbookEntries;
    if (!getNames || !createEntries) {
      console.warn('[开场界面] createWorldbookEntries API 不可用，跳过');
      return;
    }

    const books = getNames('current');
    if (!books?.primary) {
      console.warn('[开场界面] 角色卡未绑定主世界书，跳过');
      return;
    }

    const entryName = `${state.gender}·${state.kungfu}·自定义`;
    const content   = `关于{{user}}\n外貌：${state.appearance}\n性格：${state.personality}`;

    await createEntries(books.primary, [{
      name:     entryName,
      strategy: { type: 'constant', keys: [] },
      position: { type: 'before_character_definition', order: 100 },
      content:  content,
      enabled:  true,
      probability: 100,
      recursion: { prevent_incoming: true, prevent_outgoing: true, delay_until: null },
      effect:   { sticky: null, cooldown: null, delay: null },
    }]);

    showWorldbookNotice(entryName);

  } catch(e) {
    console.warn('[开场界面] 世界书写入跳过:', e.message);
  }
}

function showWorldbookNotice(entryName) {
  const overlay = document.getElementById('wb-notice-overlay');
  const text    = document.getElementById('wb-notice-text');
  if (text) text.textContent =
    `已创建世界书条目「${entryName}」（蓝灯·始终激活）。` +
    `该条目不会自动删除——若需重新开始游玩，请先在世界书中手动删除该条目，再新建对话。`;
  if (overlay) {
    setTimeout(() => { overlay.style.display = 'flex'; }, 1200);
  }
}

  const oc = OPENING_CHOICES.find(c => c.id === state.openingId);
  const msg = oc && oc.editable
    ? `根据开场设定：${state.openingCustom}，开场吧。`
    : `根据开场设定，开始叙述故事吧。`;

  if (textarea) {
    textarea.value = msg;
    textarea.dispatchEvent(new Event('input',  { bubbles: true }));
    textarea.dispatchEvent(new Event('change', { bubbles: true }));
    textarea.dispatchEvent(new KeyboardEvent('keyup', { bubbles: true }));
    textarea.focus();
  }
}

function showSuccess() {
  const overlay = document.getElementById('success-overlay');
  overlay.classList.add('show');
}
